49/// Creates an array of type `[T; N]` by repeatedly cloning a value.
50///
51/// This is the same as `[val; N]`, but it also works for types that do not
52/// implement [`Copy`].
53///
54/// The provided value will be used as an element of the resulting array and
55/// will be cloned N - 1 times to fill up the rest. If N is zero, the value
56/// will be dropped.
57///
58/// # Example
59///
60/// Creating multiple copies of a `String`:
61/// ```rust
62/// #![feature(array_repeat)]
63///
64/// use std::array;
65///
66/// let string = "Hello there!".to_string();
67/// let strings = array::repeat(string);
68/// assert_eq!(strings, ["Hello there!", "Hello there!"]);
69/// ```
70#[inline]
71#[unstable(feature = "array_repeat", issue = "126695")]
72#[cfg(not(feature = "ferrocene_certified"))]
73pub fn repeat<T: Clone, const N: usize>(val: T) -> [T; N] {
74    from_trusted_iterator(repeat_n(val, N))
75}

77/// Creates an array where each element is produced by calling `f` with
78/// that element's index while walking forward through the array.
79///
80/// This is essentially the same as writing
81/// ```text
82/// [f(0), f(1), f(2), â€¦, f(N - 2), f(N - 1)]
83/// ```
84/// and is similar to `(0..i).map(f)`, just for arrays not iterators.
85///
86/// If `N == 0`, this produces an empty array without ever calling `f`.
87///
88/// # Example
89///
90/// ```rust
91/// // type inference is helping us here, the way `from_fn` knows how many
92/// // elements to produce is the length of array down there: only arrays of
93/// // equal lengths can be compared, so the const generic parameter `N` is
94/// // inferred to be 5, thus creating array of 5 elements.
95///
96/// let array = core::array::from_fn(|i| i);
97/// // indexes are:    0  1  2  3  4
98/// assert_eq!(array, [0, 1, 2, 3, 4]);
99///
100/// let array2: [usize; 8] = core::array::from_fn(|i| i * 2);
101/// // indexes are:     0  1  2  3  4  5   6   7
102/// assert_eq!(array2, [0, 2, 4, 6, 8, 10, 12, 14]);
103///
104/// let bool_arr = core::array::from_fn::<_, 5, _>(|i| i % 2 == 0);
105/// // indexes are:       0     1      2     3      4
106/// assert_eq!(bool_arr, [true, false, true, false, true]);
107/// ```
108///
109/// You can also capture things, for example to create an array full of clones
110/// where you can't just use `[item; N]` because it's not `Copy`:
111/// ```
112/// # // TBH `array::repeat` would be better for this, but it's not stable yet.
113/// let my_string = String::from("Hello");
114/// let clones: [String; 42] = std::array::from_fn(|_| my_string.clone());
115/// assert!(clones.iter().all(|x| *x == my_string));
116/// ```
117///
118/// The array is generated in ascending index order, starting from the front
119/// and going towards the back, so you can use closures with mutable state:
120/// ```
121/// let mut state = 1;
122/// let a = std::array::from_fn(|_| { let x = state; state *= 2; x });
123/// assert_eq!(a, [1, 2, 4, 8, 16, 32]);
124/// ```
125#[inline]
126#[stable(feature = "array_from_fn", since = "1.63.0")]
127#[cfg(not(feature = "ferrocene_certified"))]
128pub fn from_fn<T, const N: usize, F>(f: F) -> [T; N]
129where
130    F: FnMut(usize) -> T,
131{
132    try_from_fn(NeverShortCircuit::wrap_mut_1(f)).0
133}

135/// Creates an array `[T; N]` where each fallible array element `T` is returned by the `cb` call.
136/// Unlike [`from_fn`], where the element creation can't fail, this version will return an error
137/// if any element creation was unsuccessful.
138///
139/// The return type of this function depends on the return type of the closure.
140/// If you return `Result<T, E>` from the closure, you'll get a `Result<[T; N], E>`.
141/// If you return `Option<T>` from the closure, you'll get an `Option<[T; N]>`.
142///
143/// # Arguments
144///
145/// * `cb`: Callback where the passed argument is the current array index.
146///
147/// # Example
148///
149/// ```rust
150/// #![feature(array_try_from_fn)]
151///
152/// let array: Result<[u8; 5], _> = std::array::try_from_fn(|i| i.try_into());
153/// assert_eq!(array, Ok([0, 1, 2, 3, 4]));
154///
155/// let array: Result<[i8; 200], _> = std::array::try_from_fn(|i| i.try_into());
156/// assert!(array.is_err());
157///
158/// let array: Option<[_; 4]> = std::array::try_from_fn(|i| i.checked_add(100));
159/// assert_eq!(array, Some([100, 101, 102, 103]));
160///
161/// let array: Option<[_; 4]> = std::array::try_from_fn(|i| i.checked_sub(100));
162/// assert_eq!(array, None);
163/// ```
164#[inline]
165#[unstable(feature = "array_try_from_fn", issue = "89379")]
166#[cfg(not(feature = "ferrocene_certified"))]
167pub fn try_from_fn<R, const N: usize, F>(cb: F) -> ChangeOutputType<R, [R::Output; N]>
168where
169    F: FnMut(usize) -> R,
170    R: Try,
171    R::Residual: Residual<[R::Output; N]>,
172{
173    let mut array = [const { MaybeUninit::uninit() }; N];
174    match try_from_fn_erased(&mut array, cb) {
175        ControlFlow::Break(r) => FromResidual::from_residual(r),
176        ControlFlow::Continue(()) => {
177            // SAFETY: All elements of the array were populated.
178            try { unsafe { MaybeUninit::array_assume_init(array) } }
179        }
180    }
181}

183/// Converts a reference to `T` into a reference to an array of length 1 (without copying).
184#[stable(feature = "array_from_ref", since = "1.53.0")]
185#[rustc_const_stable(feature = "const_array_from_ref_shared", since = "1.63.0")]
186#[cfg(not(feature = "ferrocene_certified"))]
187pub const fn from_ref<T>(s: &T) -> &[T; 1] {
188    // SAFETY: Converting `&T` to `&[T; 1]` is sound.
189    unsafe { &*(s as *const T).cast::<[T; 1]>() }

534            fn default() -> [T; $n] { [] }
535        }

542impl<T, const N: usize> [T; N] {
543    /// Returns an array of the same size as `self`, with function `f` applied to each element
544    /// in order.
545    ///
546    /// If you don't necessarily need a new fixed-size array, consider using
547    /// [`Iterator::map`] instead.
548    ///
549    ///
550    /// # Note on performance and stack usage
551    ///
552    /// Unfortunately, usages of this method are currently not always optimized
553    /// as well as they could be. This mainly concerns large arrays, as mapping
554    /// over small arrays seem to be optimized just fine. Also note that in
555    /// debug mode (i.e. without any optimizations), this method can use a lot
556    /// of stack space (a few times the size of the array or more).
557    ///
558    /// Therefore, in performance-critical code, try to avoid using this method
559    /// on large arrays or check the emitted code. Also try to avoid chained
560    /// maps (e.g. `arr.map(...).map(...)`).
561    ///
562    /// In many cases, you can instead use [`Iterator::map`] by calling `.iter()`
563    /// or `.into_iter()` on your array. `[T; N]::map` is only necessary if you
564    /// really need a new array of the same size as the result. Rust's lazy

588        F: FnMut(T) -> U,

647    /// Borrows each element and returns an array of references with the same
648    /// size as `self`.
649    ///
650    ///
651    /// # Example
652    ///
653    /// ```
654    /// let floats = [3.1, 2.7, -1.0];
655    /// let float_refs: [&f64; 3] = floats.each_ref();
656    /// assert_eq!(float_refs, [&3.1, &2.7, &-1.0]);
657    /// ```
658    ///
659    /// This method is particularly useful if combined with other methods, like
660    /// [`map`](#method.map). This way, you can avoid moving the original
661    /// array if its elements are not [`Copy`].
662    ///
663    /// ```
664    /// let strings = ["Ferris".to_string(), "â™¥".to_string(), "Rust".to_string()];
665    /// let is_ascii = strings.each_ref().map(|s| s.is_ascii());
666    /// assert_eq!(is_ascii, [true, false, true]);
667    ///
668    /// // We can still access the original array: it has not been moved.
669    /// assert_eq!(strings.len(), 3);
670    /// ```
671    #[stable(feature = "array_methods", since = "1.77.0")]
672    #[rustc_const_unstable(feature = "const_array_each_ref", issue = "133289")]
673    #[cfg(not(feature = "ferrocene_certified"))]
674    pub const fn each_ref(&self) -> [&T; N] {
675        let mut buf = [null::<T>(); N];
676
677        // FIXME(const-hack): We would like to simply use iterators for this (as in the original implementation), but this is not allowed in constant expressions.
678        let mut i = 0;
679        while i < N {
680            buf[i] = &raw const self[i];
681
682            i += 1;
683        }
684
685        // SAFETY: `*const T` has the same layout as `&T`, and we've also initialised each pointer as a valid reference.
686        unsafe { transmute_unchecked(buf) }
687    }

882/// Populate an array from the first `N` elements of `iter`
883///
884/// # Panics
885///
886/// If the iterator doesn't actually have enough items.
887///
888/// By depending on `TrustedLen`, however, we can do that check up-front (where
889/// it easily optimizes away) so it doesn't impact the loop that fills the array.
890#[inline]
891#[cfg(not(feature = "ferrocene_certified"))]
892fn from_trusted_iterator<T, const N: usize>(iter: impl UncheckedIterator<Item = T>) -> [T; N] {
893    try_from_trusted_iterator(iter.map(NeverShortCircuit)).0
894}
895
896#[inline]
897#[cfg(not(feature = "ferrocene_certified"))]
898fn try_from_trusted_iterator<T, R, const N: usize>(
899    iter: impl UncheckedIterator<Item = R>,
900) -> ChangeOutputType<R, [T; N]>
901where
902    R: Try<Output = T>,
903    R::Residual: Residual<[T; N]>,
904{
905    assert!(iter.size_hint().0 >= N);
906    fn next<T>(mut iter: impl UncheckedIterator<Item = T>) -> impl FnMut(usize) -> T {
907        move |_| {
908            // SAFETY: We know that `from_fn` will call this at most N times,
909            // and we checked to ensure that we have at least that many items.
910            unsafe { iter.next_unchecked() }
911        }
912    }
913
914    try_from_fn(next(iter))
915}

917/// Version of [`try_from_fn`] using a passed-in slice in order to avoid
918/// needing to monomorphize for every array length.
919///
920/// This takes a generator rather than an iterator so that *at the type level*
921/// it never needs to worry about running out of items.  When combined with
922/// an infallible `Try` type, that means the loop canonicalizes easily, allowing
923/// it to optimize well.
924///
925/// It would be *possible* to unify this and [`iter_next_chunk_erased`] into one
926/// function that does the union of both things, but last time it was that way
927/// it resulted in poor codegen from the "are there enough source items?" checks
928/// not optimizing away.  So if you give it a shot, make sure to watch what
929/// happens in the codegen tests.
930#[inline]
931#[cfg(not(feature = "ferrocene_certified"))]
932fn try_from_fn_erased<T, R>(
933    buffer: &mut [MaybeUninit<T>],
934    mut generator: impl FnMut(usize) -> R,
935) -> ControlFlow<R::Residual>
936where
937    R: Try<Output = T>,
938{
939    let mut guard = Guard { array_mut: buffer, initialized: 0 };
940
941    while guard.initialized < guard.array_mut.len() {
942        let item = generator(guard.initialized).branch()?;
943
944        // SAFETY: The loop condition ensures we have space to push the item
945        unsafe { guard.push_unchecked(item) };
946    }
947
948    mem::forget(guard);
949    ControlFlow::Continue(())
950}

952/// Panic guard for incremental initialization of arrays.
953///
954/// Disarm the guard with `mem::forget` once the array has been initialized.
955///
956/// # Safety
957///
958/// All write accesses to this structure are unsafe and must maintain a correct
959/// count of `initialized` elements.
960///
961/// To minimize indirection fields are still pub but callers should at least use
962/// `push_unchecked` to signal that something unsafe is going on.
963#[cfg(not(feature = "ferrocene_certified"))]
964struct Guard<'a, T> {
965    /// The array to be initialized.
966    pub array_mut: &'a mut [MaybeUninit<T>],
967    /// The number of items that have been initialized so far.
968    pub initialized: usize,
969}
970
971#[cfg(not(feature = "ferrocene_certified"))]
972impl<T> Guard<'_, T> {
973    /// Adds an item to the array and updates the initialized item counter.
974    ///
975    /// # Safety
976    ///
977    /// No more than N elements must be initialized.
978    #[inline]
979    pub(crate) unsafe fn push_unchecked(&mut self, item: T) {
980        // SAFETY: If `initialized` was correct before and the caller does not
981        // invoke this method more than N times then writes will be in-bounds
982        // and slots will not be initialized more than once.
983        unsafe {
984            self.array_mut.get_unchecked_mut(self.initialized).write(item);
985            self.initialized = self.initialized.unchecked_add(1);
986        }
987    }
988}
989
990#[cfg(not(feature = "ferrocene_certified"))]
991impl<T> Drop for Guard<'_, T> {
992    #[inline]
993    fn drop(&mut self) {
994        debug_assert!(self.initialized <= self.array_mut.len());
995
996        // SAFETY: this slice will contain only initialized objects.
997        unsafe {
998            self.array_mut.get_unchecked_mut(..self.initialized).assume_init_drop();
999        }
1000    }
1001}

1003/// Pulls `N` items from `iter` and returns them as an array. If the iterator
1004/// yields fewer than `N` items, `Err` is returned containing an iterator over
1005/// the already yielded items.
1006///
1007/// Since the iterator is passed as a mutable reference and this function calls
1008/// `next` at most `N` times, the iterator can still be used afterwards to
1009/// retrieve the remaining items.
1010///
1011/// If `iter.next()` panicks, all items already yielded by the iterator are
1012/// dropped.
1013///
1014/// Used for [`Iterator::next_chunk`].
1015#[inline]
1016#[cfg(not(feature = "ferrocene_certified"))]
1017pub(crate) fn iter_next_chunk<T, const N: usize>(
1018    iter: &mut impl Iterator<Item = T>,
1019) -> Result<[T; N], IntoIter<T, N>> {
1020    let mut array = [const { MaybeUninit::uninit() }; N];
1021    let r = iter_next_chunk_erased(&mut array, iter);
1022    match r {
1023        Ok(()) => {
1024            // SAFETY: All elements of `array` were populated.
1025            Ok(unsafe { MaybeUninit::array_assume_init(array) })
1026        }
1027        Err(initialized) => {
1028            // SAFETY: Only the first `initialized` elements were populated
1029            Err(unsafe { IntoIter::new_unchecked(array, 0..initialized) })
1030        }
1031    }
1032}

1034/// Version of [`iter_next_chunk`] using a passed-in slice in order to avoid
1035/// needing to monomorphize for every array length.
1036///
1037/// Unfortunately this loop has two exit conditions, the buffer filling up
1038/// or the iterator running out of items, making it tend to optimize poorly.
1039#[inline]
1040#[cfg(not(feature = "ferrocene_certified"))]
1041fn iter_next_chunk_erased<T>(
1042    buffer: &mut [MaybeUninit<T>],
1043    iter: &mut impl Iterator<Item = T>,
1044) -> Result<(), usize> {
1045    let mut guard = Guard { array_mut: buffer, initialized: 0 };
1046    while guard.initialized < guard.array_mut.len() {
1047        let Some(item) = iter.next() else {
1048            // Unlike `try_from_fn_erased`, we want to keep the partial results,
1049            // so we need to defuse the guard instead of using `?`.
1050            let initialized = guard.initialized;
1051            mem::forget(guard);
1052            return Err(initialized);
1053        };
1054
1055        // SAFETY: The loop condition ensures we have space to push the item
1056        unsafe { guard.push_unchecked(item) };
1057    }
1058
1059    mem::forget(guard);
1060    Ok(())
