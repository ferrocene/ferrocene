1//! The `Clone` trait for types that cannot be 'implicitly copied'.
2//!
3//! In Rust, some simple types are "implicitly copyable" and when you
4//! assign them or pass them as arguments, the receiver will get a copy,
5//! leaving the original value in place. These types do not require
6//! allocation to copy and do not have finalizers (i.e., they do not
7//! contain owned boxes or implement [`Drop`]), so the compiler considers
8//! them cheap and safe to copy. For other types copies must be made
9//! explicitly, by convention implementing the [`Clone`] trait and calling
10//! the [`clone`] method.
11//!
12//! [`clone`]: Clone::clone
13//!
14//! Basic usage example:
15//!
16//! ```
17//! let s = String::new(); // String type implements Clone
18//! let copy = s.clone(); // so we can clone it
19//! ```
20//!
21//! To easily implement the Clone trait, you can also use
22//! `#[derive(Clone)]`. Example:
23//!
24//! ```
25//! #[derive(Clone)] // we add the Clone trait to Morpheus struct
26//! struct Morpheus {
27//!    blue_pill: f32,
28//!    red_pill: i64,
29//! }
30//!
31//! fn main() {
32//!    let f = Morpheus { blue_pill: 0.0, red_pill: 0 };
33//!    let copy = f.clone(); // and now we can clone it!
34//! }
35//! ```

42/// A common trait for the ability to explicitly duplicate an object.
43///
44/// Differs from [`Copy`] in that [`Copy`] is implicit and an inexpensive bit-wise copy, while
45/// `Clone` is always explicit and may or may not be expensive. In order to enforce
46/// these characteristics, Rust does not allow you to reimplement [`Copy`], but you
47/// may reimplement `Clone` and run arbitrary code.
48///
49/// Since `Clone` is more general than [`Copy`], you can automatically make anything
50/// [`Copy`] be `Clone` as well.
51///
52/// ## Derivable
53///
54/// This trait can be used with `#[derive]` if all fields are `Clone`. The `derive`d
55/// implementation of [`Clone`] calls [`clone`] on each field.
56///
57/// [`clone`]: Clone::clone
58///
59/// For a generic struct, `#[derive]` implements `Clone` conditionally by adding bound `Clone` on
60/// generic parameters.
61///
62/// ```
63/// // `derive` implements Clone for Reading<T> when T is Clone.
64/// #[derive(Clone)]
65/// struct Reading<T> {
66///     frequency: T,
67/// }
68/// ```
69///
70/// ## How can I implement `Clone`?
71///
72/// Types that are [`Copy`] should have a trivial implementation of `Clone`. More formally:
73/// if `T: Copy`, `x: T`, and `y: &T`, then `let x = y.clone();` is equivalent to `let x = *y;`.
74/// Manual implementations should be careful to uphold this invariant; however, unsafe code
75/// must not rely on it to ensure memory safety.
76///
77/// An example is a generic struct holding a function pointer. In this case, the
78/// implementation of `Clone` cannot be `derive`d, but can be implemented as:
79///
80/// ```
81/// struct Generate<T>(fn() -> T);
82///
83/// impl<T> Copy for Generate<T> {}
84///
85/// impl<T> Clone for Generate<T> {
86///     fn clone(&self) -> Self {
87///         *self
88///     }
89/// }
90/// ```
91///
92/// If we `derive`:
93///
94/// ```
95/// #[derive(Copy, Clone)]
96/// struct Generate<T>(fn() -> T);
97/// ```
98///
99/// the auto-derived implementations will have unnecessary `T: Copy` and `T: Clone` bounds:
100///
101/// ```
102/// # struct Generate<T>(fn() -> T);
103///
104/// // Automatically derived
105/// impl<T: Copy> Copy for Generate<T> { }
106///
107/// // Automatically derived
108/// impl<T: Clone> Clone for Generate<T> {
109///     fn clone(&self) -> Generate<T> {
110///         Generate(Clone::clone(&self.0))
111///     }
112/// }
113/// ```
114///
115/// The bounds are unnecessary because clearly the function itself should be
116/// copy- and cloneable even if its return type is not:
117///
118/// ```compile_fail,E0599
119/// #[derive(Copy, Clone)]
120/// struct Generate<T>(fn() -> T);
121///
122/// struct NotCloneable;
123///
124/// fn generate_not_cloneable() -> NotCloneable {
125///     NotCloneable
126/// }
127///
128/// Generate(generate_not_cloneable).clone(); // error: trait bounds were not satisfied
129/// // Note: With the manual implementations the above line will compile.
130/// ```
131///
132/// ## Additional implementors
133///
134/// In addition to the [implementors listed below][impls],
135/// the following types also implement `Clone`:
136///
137/// * Function item types (i.e., the distinct types defined for each function)
138/// * Function pointer types (e.g., `fn() -> i32`)
139/// * Closure types, if they capture no value from the environment
140///   or if all such captured values implement `Clone` themselves.
141///   Note that variables captured by shared reference always implement `Clone`
142///   (even if the referent doesn't),
143///   while variables captured by mutable reference never implement `Clone`.
144///
145/// [impls]: #implementors
146#[stable(feature = "rust1", since = "1.0.0")]
147#[lang = "clone"]
148#[rustc_diagnostic_item = "Clone"]
149#[rustc_trivial_field_reads]
150pub trait Clone: Sized {
151    /// Returns a copy of the value.
152    ///
153    /// # Examples
154    ///
155    /// ```
156    /// # #![allow(noop_method_call)]
157    /// let hello = "Hello"; // &str implements Clone
158    ///
159    /// assert_eq!("Hello", hello.clone());
160    /// ```
161    #[stable(feature = "rust1", since = "1.0.0")]
162    #[must_use = "cloning is often expensive and is not expected to have side effects"]
163    // Clone::clone is special because the compiler generates MIR to implement it for some types.
164    // See InstanceKind::CloneShim.
165    #[lang = "clone_fn"]
166    fn clone(&self) -> Self;
167
168    /// Performs copy-assignment from `source`.
169    ///
170    /// `a.clone_from(&b)` is equivalent to `a = b.clone()` in functionality,
171    /// but can be overridden to reuse the resources of `a` to avoid unnecessary
172    /// allocations.
173    #[inline]
174    #[stable(feature = "rust1", since = "1.0.0")]
175    fn clone_from(&mut self, source: &Self) {
176        *self = source.clone()
177    }
178}

188/// Trait for objects whose [`Clone`] impl is lightweight (e.g. reference-counted)
189///
190/// Cloning an object implementing this trait should in general:
191/// - be O(1) (constant) time regardless of the amount of data managed by the object,
192/// - not require a memory allocation,
193/// - not require copying more than roughly 64 bytes (a typical cache line size),
194/// - not block the current thread,
195/// - not have any semantic side effects (e.g. allocating a file descriptor), and
196/// - not have overhead larger than a couple of atomic operations.
197///
198/// The `UseCloned` trait does not provide a method; instead, it indicates that
199/// `Clone::clone` is lightweight, and allows the use of the `.use` syntax.
200///
201/// ## .use postfix syntax
202///
203/// Values can be `.use`d by adding `.use` postfix to the value you want to use.
204///
205/// ```ignore (this won't work until we land use)
206/// fn foo(f: Foo) {
207///     // if `Foo` implements `Copy` f would be copied into x.
208///     // if `Foo` implements `UseCloned` f would be cloned into x.
209///     // otherwise f would be moved into x.
210///     let x = f.use;
211///     // ...
212/// }
213/// ```
214///
215/// ## use closures
216///
217/// Use closures allow captured values to be automatically used.
218/// This is similar to have a closure that you would call `.use` over each captured value.
219#[unstable(feature = "ergonomic_clones", issue = "132290")]
220#[lang = "use_cloned"]
221#[cfg(not(feature = "ferrocene_certified"))]
222pub trait UseCloned: Clone {
223    // Empty.
224}

270/// A generalization of [`Clone`] to [dynamically-sized types][DST] stored in arbitrary containers.
271///
272/// This trait is implemented for all types implementing [`Clone`], [slices](slice) of all
273/// such types, and other dynamically-sized types in the standard library.
274/// You may also implement this trait to enable cloning custom DSTs
275/// (structures containing dynamically-sized fields), or use it as a supertrait to enable
276/// cloning a [trait object].
277///
278/// This trait is normally used via operations on container types which support DSTs,
279/// so you should not typically need to call `.clone_to_uninit()` explicitly except when
280/// implementing such a container or otherwise performing explicit management of an allocation,
281/// or when implementing `CloneToUninit` itself.
282///
283/// # Safety
284///
285/// Implementations must ensure that when `.clone_to_uninit(dest)` returns normally rather than
286/// panicking, it always leaves `*dest` initialized as a valid value of type `Self`.
287///
288/// # Examples
289///
290// FIXME(#126799): when `Box::clone` allows use of `CloneToUninit`, rewrite these examples with it
291// since `Rc` is a distraction.
292///
293/// If you are defining a trait, you can add `CloneToUninit` as a supertrait to enable cloning of
294/// `dyn` values of your trait:
295///
296/// ```
297/// #![feature(clone_to_uninit)]
298/// use std::rc::Rc;
299///
300/// trait Foo: std::fmt::Debug + std::clone::CloneToUninit {
301///     fn modify(&mut self);
302///     fn value(&self) -> i32;
303/// }
304///
305/// impl Foo for i32 {
306///     fn modify(&mut self) {
307///         *self *= 10;
308///     }
309///     fn value(&self) -> i32 {
310///         *self
311///     }
312/// }
313///
314/// let first: Rc<dyn Foo> = Rc::new(1234);
315///
316/// let mut second = first.clone();
317/// Rc::make_mut(&mut second).modify(); // make_mut() will call clone_to_uninit()
318///
319/// assert_eq!(first.value(), 1234);
320/// assert_eq!(second.value(), 12340);
321/// ```
322///
323/// The following is an example of implementing `CloneToUninit` for a custom DST.
324/// (It is essentially a limited form of what `derive(CloneToUninit)` would do,
325/// if such a derive macro existed.)
326///
327/// ```
328/// #![feature(clone_to_uninit)]
329/// use std::clone::CloneToUninit;
330/// use std::mem::offset_of;
331/// use std::rc::Rc;
332///
333/// #[derive(PartialEq)]
334/// struct MyDst<T: ?Sized> {
335///     label: String,
336///     contents: T,
337/// }
338///
339/// unsafe impl<T: ?Sized + CloneToUninit> CloneToUninit for MyDst<T> {
340///     unsafe fn clone_to_uninit(&self, dest: *mut u8) {
341///         // The offset of `self.contents` is dynamic because it depends on the alignment of T
342///         // which can be dynamic (if `T = dyn SomeTrait`). Therefore, we have to obtain it
343///         // dynamically by examining `self`, rather than using `offset_of!`.
344///         //
345///         // SAFETY: `self` by definition points somewhere before `&self.contents` in the same
346///         // allocation.
347///         let offset_of_contents = unsafe {
348///             (&raw const self.contents).byte_offset_from_unsigned(self)
349///         };
350///
351///         // Clone the *sized* fields of `self` (just one, in this example).
352///         // (By cloning this first and storing it temporarily in a local variable, we avoid
353///         // leaking it in case of any panic, using the ordinary automatic cleanup of local
354///         // variables. Such a leak would be sound, but undesirable.)
355///         let label = self.label.clone();
356///
357///         // SAFETY: The caller must provide a `dest` such that these field offsets are valid
358///         // to write to.
359///         unsafe {
360///             // Clone the unsized field directly from `self` to `dest`.
361///             self.contents.clone_to_uninit(dest.add(offset_of_contents));
362///
363///             // Now write all the sized fields.
364///             //
365///             // Note that we only do this once all of the clone() and clone_to_uninit() calls
366///             // have completed, and therefore we know that there are no more possible panics;
367///             // this ensures no memory leaks in case of panic.
368///             dest.add(offset_of!(Self, label)).cast::<String>().write(label);
369///         }
370///         // All fields of the struct have been initialized; therefore, the struct is initialized,
371///         // and we have satisfied our `unsafe impl CloneToUninit` obligations.
372///     }
373/// }
374///
375/// fn main() {
376///     // Construct MyDst<[u8; 4]>, then coerce to MyDst<[u8]>.
377///     let first: Rc<MyDst<[u8]>> = Rc::new(MyDst {
378///         label: String::from("hello"),
379///         contents: [1, 2, 3, 4],
380///     });
381///
382///     let mut second = first.clone();
383///     // make_mut() will call clone_to_uninit().
384///     for elem in Rc::make_mut(&mut second).contents.iter_mut() {
385///         *elem *= 10;
386///     }
387///
388///     assert_eq!(first.contents, [1, 2, 3, 4]);
389///     assert_eq!(second.contents, [10, 20, 30, 40]);
390///     assert_eq!(second.label, "hello");
391/// }
392/// ```
393///
394/// # See Also
395///
396/// * [`Clone::clone_from`] is a safe function which may be used instead when [`Self: Sized`](Sized)
397///   and the destination is already initialized; it may be able to reuse allocations owned by
398///   the destination, whereas `clone_to_uninit` cannot, since its destination is assumed to be
399///   uninitialized.
400/// * [`ToOwned`], which allocates a new destination container.
401///
402/// [`ToOwned`]: ../../std/borrow/trait.ToOwned.html
403/// [DST]: https://doc.rust-lang.org/reference/dynamically-sized-types.html
404/// [trait object]: https://doc.rust-lang.org/reference/types/trait-object.html
405#[unstable(feature = "clone_to_uninit", issue = "126799")]
406#[cfg(not(feature = "ferrocene_certified"))]
407pub unsafe trait CloneToUninit {
408    /// Performs copy-assignment from `self` to `dest`.
409    ///
410    /// This is analogous to `std::ptr::write(dest.cast(), self.clone())`,
411    /// except that `Self` may be a dynamically-sized type ([`!Sized`](Sized)).
412    ///
413    /// Before this function is called, `dest` may point to uninitialized memory.
414    /// After this function is called, `dest` will point to initialized memory; it will be
415    /// sound to create a `&Self` reference from the pointer with the [pointer metadata]
416    /// from `self`.
417    ///
418    /// # Safety
419    ///
420    /// Behavior is undefined if any of the following conditions are violated:
421    ///
422    /// * `dest` must be [valid] for writes for `size_of_val(self)` bytes.
423    /// * `dest` must be properly aligned to `align_of_val(self)`.
424    ///
425    /// [valid]: crate::ptr#safety
426    /// [pointer metadata]: crate::ptr::metadata()
427    ///
428    /// # Panics
429    ///
430    /// This function may panic. (For example, it might panic if memory allocation for a clone
431    /// of a value owned by `self` fails.)
432    /// If the call panics, then `*dest` should be treated as uninitialized memory; it must not be
433    /// read or dropped, because even if it was previously valid, it may have been partially
434    /// overwritten.
435    ///
436    /// The caller may wish to take care to deallocate the allocation pointed to by `dest`,
437    /// if applicable, to avoid a memory leak (but this is not a requirement).
438    ///
439    /// Implementors should avoid leaking values by, upon unwinding, dropping all component values
440    /// that might have already been created. (For example, if a `[Foo]` of length 3 is being
441    /// cloned, and the second of the three calls to `Foo::clone()` unwinds, then the first `Foo`
442    /// cloned should be dropped.)
443    unsafe fn clone_to_uninit(&self, dest: *mut u8);
444}

446#[unstable(feature = "clone_to_uninit", issue = "126799")]
447#[cfg(not(feature = "ferrocene_certified"))]
448unsafe impl<T: Clone> CloneToUninit for T {
449    #[inline]
450    unsafe fn clone_to_uninit(&self, dest: *mut u8) {
451        // SAFETY: we're calling a specialization with the same contract
452        unsafe { <T as self::uninit::CopySpec>::clone_one(self, dest.cast::<T>()) }
453    }
454}

456#[unstable(feature = "clone_to_uninit", issue = "126799")]
457#[cfg(not(feature = "ferrocene_certified"))]
458unsafe impl<T: Clone> CloneToUninit for [T] {
459    #[inline]
460    #[cfg_attr(debug_assertions, track_caller)]
461    unsafe fn clone_to_uninit(&self, dest: *mut u8) {
462        let dest: *mut [T] = dest.with_metadata_of(self);
463        // SAFETY: we're calling a specialization with the same contract
464        unsafe { <T as self::uninit::CopySpec>::clone_slice(self, dest) }
465    }
466}
