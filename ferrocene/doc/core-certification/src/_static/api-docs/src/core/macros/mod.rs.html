16///
17/// Assertions are always checked in both debug and release builds, and cannot
18/// be disabled. See [`debug_assert_eq!`] for assertions that are disabled in
19/// release builds by default.
20///
21/// [`debug_assert_eq!`]: crate::debug_assert_eq
22///
23/// On panic, this macro will print the values of the expressions with their
24/// debug representations.
25///
26/// Like [`assert!`], this macro has a second form, where a custom
27/// panic message can be provided.
28///
29/// # Examples
30///
31/// ```
32/// let a = 3;
33/// let b = 1 + 2;
34/// assert_eq!(a, b);
35///
36/// assert_eq!(a, b, "we are testing addition with {} and {}", a, b);
37/// ```
38#[macro_export]
39#[stable(feature = "rust1", since = "1.0.0")]
40#[rustc_diagnostic_item = "assert_eq_macro"]
41#[allow_internal_unstable(panic_internals)]
42#[cfg(not(feature = "ferrocene_certified"))]
43macro_rules! assert_eq {
44    ($left:expr, $right:expr $(,)?) => {
45        match (&$left, &$right) {
46            (left_val, right_val) => {
47                if !(*left_val == *right_val) {
48                    let kind = $crate::panicking::AssertKind::Eq;
49                    // The reborrows below are intentional. Without them, the stack slot for the
50                    // borrow is initialized even before the values are compared, leading to a
51                    // noticeable slow down.
52                    $crate::panicking::assert_failed(kind, &*left_val, &*right_val, $crate::option::Option::None);
53                }
54            }
55        }
56    };
57    ($left:expr, $right:expr, $($arg:tt)+) => {
58        match (&$left, &$right) {
59            (left_val, right_val) => {
60                if !(*left_val == *right_val) {
61                    let kind = $crate::panicking::AssertKind::Eq;
62                    // The reborrows below are intentional. Without them, the stack slot for the
63                    // borrow is initialized even before the values are compared, leading to a
64                    // noticeable slow down.
65                    $crate::panicking::assert_failed(kind, &*left_val, &*right_val, $crate::option::Option::Some($crate::format_args!($($arg)+)));
66                }
67            }
68        }

129/// Asserts that an expression matches the provided pattern.
130///
131/// This macro is generally preferable to `assert!(matches!(value, pattern))`, because it can print
132/// the debug representation of the actual value shape that did not meet expectations. In contrast,
133/// using [`assert!`] will only print that expectations were not met, but not why.
134///
135/// The pattern syntax is exactly the same as found in a match arm and the `matches!` macro. The
136/// optional if guard can be used to add additional checks that must be true for the matched value,
137/// otherwise this macro will panic.
138///
139/// Assertions are always checked in both debug and release builds, and cannot
140/// be disabled. See [`debug_assert_matches!`] for assertions that are disabled in
141/// release builds by default.
142///
143/// [`debug_assert_matches!`]: crate::assert_matches::debug_assert_matches
144///
145/// On panic, this macro will print the value of the expression with its debug representation.
146///
147/// Like [`assert!`], this macro has a second form, where a custom panic message can be provided.
148///
149/// # Examples
150///
151/// ```
152/// #![feature(assert_matches)]
153///
154/// use std::assert_matches::assert_matches;
155///
156/// let a = Some(345);
157/// let b = Some(56);
158/// assert_matches!(a, Some(_));
159/// assert_matches!(b, Some(_));
160///
161/// assert_matches!(a, Some(345));
162/// assert_matches!(a, Some(345) | None);
163///
164/// // assert_matches!(a, None); // panics
165/// // assert_matches!(b, Some(345)); // panics
166/// // assert_matches!(b, Some(345) | None); // panics
167///
168/// assert_matches!(a, Some(x) if x > 100);
169/// // assert_matches!(a, Some(x) if x < 100); // panics
170/// ```
171#[unstable(feature = "assert_matches", issue = "82775")]
172#[allow_internal_unstable(panic_internals)]
173#[rustc_macro_transparency = "semitransparent"]
174#[cfg(not(feature = "ferrocene_certified"))]
175pub macro assert_matches {
176    ($left:expr, $(|)? $( $pattern:pat_param )|+ $( if $guard: expr )? $(,)?) => {
177        match $left {
178            $( $pattern )|+ $( if $guard )? => {}
179            ref left_val => {
180                $crate::panicking::assert_matches_failed(
181                    left_val,
182                    $crate::stringify!($($pattern)|+ $(if $guard)?),
183                    $crate::option::Option::None
184                );
185            }
186        }
187    },
188    ($left:expr, $(|)? $( $pattern:pat_param )|+ $( if $guard: expr )?, $($arg:tt)+) => {
189        match $left {
190            $( $pattern )|+ $( if $guard )? => {}
191            ref left_val => {
192                $crate::panicking::assert_matches_failed(
193                    left_val,
194                    $crate::stringify!($($pattern)|+ $(if $guard)?),
195                    $crate::option::Option::Some($crate::format_args!($($arg)+))
196                );
197            }
198        }
199    },
200}

247/// Asserts that a boolean expression is `true` at runtime.
248///
249/// This will invoke the [`panic!`] macro if the provided expression cannot be
250/// evaluated to `true` at runtime.
251///
252/// Like [`assert!`], this macro also has a second version, where a custom panic
253/// message can be provided.
254///
255/// # Uses
256///
257/// Unlike [`assert!`], `debug_assert!` statements are only enabled in non
258/// optimized builds by default. An optimized build will not execute
259/// `debug_assert!` statements unless `-C debug-assertions` is passed to the
260/// compiler. This makes `debug_assert!` useful for checks that are too
261/// expensive to be present in a release build but may be helpful during
262/// development. The result of expanding `debug_assert!` is always type checked.
263///
264/// An unchecked assertion allows a program in an inconsistent state to keep
265/// running, which might have unexpected consequences but does not introduce
266/// unsafety as long as this only happens in safe code. The performance cost
267/// of assertions, however, is not measurable in general. Replacing [`assert!`]
268/// with `debug_assert!` is thus only encouraged after thorough profiling, and
269/// more importantly, only in safe code!
270///
271/// # Examples
272///
273/// ```
274/// // the panic message for these assertions is the stringified value of the
275/// // expression given.
276/// debug_assert!(true);
277///
278/// fn some_expensive_computation() -> bool { true } // a very simple function
279/// debug_assert!(some_expensive_computation());
280///
281/// // assert with a custom message
282/// let x = true;
283/// debug_assert!(x, "x wasn't true!");
284///
285/// let a = 3; let b = 27;
286/// debug_assert!(a + b == 30, "a = {}, b = {}", a, b);
287/// ```
288#[macro_export]
289#[stable(feature = "rust1", since = "1.0.0")]
290#[rustc_diagnostic_item = "debug_assert_macro"]
291#[allow_internal_unstable(edition_panic)]
292#[cfg(not(feature = "ferrocene_certified"))]
293macro_rules! debug_assert {
294    ($($arg:tt)*) => {
295        if $crate::cfg!(debug_assertions) {
296            $crate::assert!($($arg)*);
297        }
298    };
299}

446/// Unwraps a result or propagates its error.
447///
448/// The [`?` operator][propagating-errors] was added to replace `try!`
449/// and should be used instead. Furthermore, `try` is a reserved word
450/// in Rust 2018, so if you must use it, you will need to use the
451/// [raw-identifier syntax][ris]: `r#try`.
452///
453/// [propagating-errors]: https://doc.rust-lang.org/book/ch09-02-recoverable-errors-with-result.html#a-shortcut-for-propagating-errors-the--operator
454/// [ris]: https://doc.rust-lang.org/nightly/rust-by-example/compatibility/raw_identifiers.html
455///
456/// `try!` matches the given [`Result`]. In case of the `Ok` variant, the
457/// expression has the value of the wrapped value.
458///
459/// In case of the `Err` variant, it retrieves the inner error. `try!` then
460/// performs conversion using `From`. This provides automatic conversion
461/// between specialized errors and more general ones. The resulting
462/// error is then immediately returned.
463///
464/// Because of the early return, `try!` can only be used in functions that
465/// return [`Result`].
466///
467/// # Examples
468///
469/// ```
470/// use std::io;
471/// use std::fs::File;
472/// use std::io::prelude::*;
473///
474/// enum MyError {
475///     FileWriteError
476/// }
477///
478/// impl From<io::Error> for MyError {
479///     fn from(e: io::Error) -> MyError {
480///         MyError::FileWriteError
481///     }
482/// }
483///
484/// // The preferred method of quick returning Errors
485/// fn write_to_file_question() -> Result<(), MyError> {
486///     let mut file = File::create("my_best_friends.txt")?;
487///     file.write_all(b"This is a list of my best friends.")?;
488///     Ok(())
489/// }
490///
491/// // The previous method of quick returning Errors
492/// fn write_to_file_using_try() -> Result<(), MyError> {
493///     let mut file = r#try!(File::create("my_best_friends.txt"));
494///     r#try!(file.write_all(b"This is a list of my best friends."));
495///     Ok(())
496/// }
497///
498/// // This is equivalent to:
499/// fn write_to_file_using_match() -> Result<(), MyError> {
500///     let mut file = r#try!(File::create("my_best_friends.txt"));
501///     match file.write_all(b"This is a list of my best friends.") {
502///         Ok(v) => v,
503///         Err(e) => return Err(From::from(e)),
504///     }
505///     Ok(())
506/// }
507/// ```
508#[macro_export]
509#[stable(feature = "rust1", since = "1.0.0")]
510#[deprecated(since = "1.39.0", note = "use the `?` operator instead")]
511#[doc(alias = "?")]
512#[cfg(not(feature = "ferrocene_certified"))]
513macro_rules! r#try {
514    ($expr:expr $(,)?) => {
515        match $expr {
516            $crate::result::Result::Ok(val) => val,
517            $crate::result::Result::Err(err) => {
518                return $crate::result::Result::Err($crate::convert::From::from(err));
519            }
520        }
521    };
522}

524/// Writes formatted data into a buffer.
525///
526/// This macro accepts a 'writer', a format string, and a list of arguments. Arguments will be
527/// formatted according to the specified format string and the result will be passed to the writer.
528/// The writer may be any value with a `write_fmt` method; generally this comes from an
529/// implementation of either the [`fmt::Write`] or the [`io::Write`] trait. The macro
530/// returns whatever the `write_fmt` method returns; commonly a [`fmt::Result`], or an
531/// [`io::Result`].
532///
533/// See [`std::fmt`] for more information on the format string syntax.
534///
535/// [`std::fmt`]: ../std/fmt/index.html
536/// [`fmt::Write`]: crate::fmt::Write
537/// [`io::Write`]: ../std/io/trait.Write.html
538/// [`fmt::Result`]: crate::fmt::Result
539/// [`io::Result`]: ../std/io/type.Result.html
540///
541/// # Examples
542///
543/// ```
544/// use std::io::Write;
545///
546/// fn main() -> std::io::Result<()> {
547///     let mut w = Vec::new();
548///     write!(&mut w, "test")?;
549///     write!(&mut w, "formatted {}", "arguments")?;
550///
551///     assert_eq!(w, b"testformatted arguments");
552///     Ok(())
553/// }
554/// ```
555///
556/// A module can import both `std::fmt::Write` and `std::io::Write` and call `write!` on objects
557/// implementing either, as objects do not typically implement both. However, the module must
558/// avoid conflict between the trait names, such as by importing them as `_` or otherwise renaming
559/// them:
560///
561/// ```
562/// use std::fmt::Write as _;
563/// use std::io::Write as _;
564///
565/// fn main() -> Result<(), Box<dyn std::error::Error>> {
566///     let mut s = String::new();
567///     let mut v = Vec::new();
568///
569///     write!(&mut s, "{} {}", "abc", 123)?; // uses fmt::Write::write_fmt
570///     write!(&mut v, "s = {:?}", s)?; // uses io::Write::write_fmt
571///     assert_eq!(v, b"s = \"abc 123\"");
572///     Ok(())
573/// }
574/// ```
575///
576/// If you also need the trait names themselves, such as to implement one or both on your types,
577/// import the containing module and then name them with a prefix:
578///
579/// ```
580/// # #![allow(unused_imports)]
581/// use std::fmt::{self, Write as _};
582/// use std::io::{self, Write as _};
583///
584/// struct Example;
585///
586/// impl fmt::Write for Example {
587///     fn write_str(&mut self, _s: &str) -> core::fmt::Result {
588///          unimplemented!();
589///     }
590/// }
591/// ```
592///
593/// Note: This macro can be used in `no_std` setups as well.
594/// In a `no_std` setup you are responsible for the implementation details of the components.
595///
596/// ```no_run
597/// use core::fmt::Write;
598///
599/// struct Example;
600///
601/// impl Write for Example {
602///     fn write_str(&mut self, _s: &str) -> core::fmt::Result {
603///          unimplemented!();
604///     }
605/// }
606///
607/// let mut m = Example{};
608/// write!(&mut m, "Hello World").expect("Not written");
609/// ```
610#[macro_export]
611#[stable(feature = "rust1", since = "1.0.0")]
612#[rustc_diagnostic_item = "write_macro"]
613#[cfg(not(feature = "ferrocene_certified"))]
614macro_rules! write {
615    ($dst:expr, $($arg:tt)*) => {
616        $dst.write_fmt($crate::format_args!($($arg)*))
617    };
618}

725/// Indicates unimplemented code by panicking with a message of "not implemented".
726///
727/// This allows your code to type-check, which is useful if you are prototyping or
728/// implementing a trait that requires multiple methods which you don't plan to use all of.
729///
730/// The difference between `unimplemented!` and [`todo!`] is that while `todo!`
731/// conveys an intent of implementing the functionality later and the message is "not yet
732/// implemented", `unimplemented!` makes no such claims. Its message is "not implemented".
733///
734/// Also, some IDEs will mark `todo!`s.
735///
736/// # Panics
737///
738/// This will always [`panic!`] because `unimplemented!` is just a shorthand for `panic!` with a
739/// fixed, specific message.
740///
741/// Like `panic!`, this macro has a second form for displaying custom values.
742///
743/// [`todo!`]: crate::todo
744///
745/// # Examples
746///
747/// Say we have a trait `Foo`:
748///
749/// ```
750/// trait Foo {
751///     fn bar(&self) -> u8;
752///     fn baz(&self);
753///     fn qux(&self) -> Result<u64, ()>;
754/// }
755/// ```
756///
757/// We want to implement `Foo` for 'MyStruct', but for some reason it only makes sense
758/// to implement the `bar()` function. `baz()` and `qux()` will still need to be defined
759/// in our implementation of `Foo`, but we can use `unimplemented!` in their definitions
760/// to allow our code to compile.
761///
762/// We still want to have our program stop running if the unimplemented methods are
763/// reached.
764///
765/// ```
766/// # trait Foo {
767/// #     fn bar(&self) -> u8;
768/// #     fn baz(&self);
769/// #     fn qux(&self) -> Result<u64, ()>;
770/// # }
771/// struct MyStruct;
772///
773/// impl Foo for MyStruct {
774///     fn bar(&self) -> u8 {
775///         1 + 1
776///     }
777///
778///     fn baz(&self) {
779///         // It makes no sense to `baz` a `MyStruct`, so we have no logic here
780///         // at all.
781///         // This will display "thread 'main' panicked at 'not implemented'".
782///         unimplemented!();
783///     }
784///
785///     fn qux(&self) -> Result<u64, ()> {
786///         // We have some logic here,
787///         // We can add a message to unimplemented! to display our omission.
788///         // This will display:
789///         // "thread 'main' panicked at 'not implemented: MyStruct isn't quxable'".
790///         unimplemented!("MyStruct isn't quxable");
791///     }
792/// }
793///
794/// fn main() {
795///     let s = MyStruct;
796///     s.bar();
797/// }
798/// ```
799#[macro_export]
800#[stable(feature = "rust1", since = "1.0.0")]
801#[rustc_diagnostic_item = "unimplemented_macro"]
802#[allow_internal_unstable(panic_internals)]
803#[cfg(not(feature = "ferrocene_certified"))]
804macro_rules! unimplemented {
805    () => {
806        $crate::panicking::panic("not implemented")
807    };
808    ($($arg:tt)+) => {
809        $crate::panic!("not implemented: {}", $crate::format_args!($($arg)+))
810    };
811}
