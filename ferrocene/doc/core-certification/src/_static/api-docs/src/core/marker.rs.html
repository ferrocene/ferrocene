29/// Implements a given marker trait for multiple types at the same time.
30///
31/// The basic syntax looks like this:
32/// ```ignore private macro
33/// marker_impls! { MarkerTrait for u8, i8 }
34/// ```
35/// You can also implement `unsafe` traits
36/// ```ignore private macro
37/// marker_impls! { unsafe MarkerTrait for u8, i8 }
38/// ```
39/// Add attributes to all impls:
40/// ```ignore private macro
41/// marker_impls! {
42///     #[allow(lint)]
43///     #[unstable(feature = "marker_trait", issue = "none")]
44///     MarkerTrait for u8, i8
45/// }
46/// ```
47/// And use generics:
48/// ```ignore private macro
49/// marker_impls! {
50///     MarkerTrait for
51///         u8, i8,
52///         {T: ?Sized} *const T,
53///         {T: ?Sized} *mut T,
54///         {T: MarkerTrait} PhantomData<T>,
55///         u32,
56/// }
57/// ```
58#[unstable(feature = "internal_impls_macro", issue = "none")]
59// Allow implementations of `UnsizedConstParamTy` even though std cannot use that feature.
60#[allow_internal_unstable(unsized_const_params)]
61macro marker_impls {
62    ( $(#[$($meta:tt)*])* $Trait:ident for $({$($bounds:tt)*})? $T:ty $(, $($rest:tt)*)? ) => {
63        $(#[$($meta)*])* impl< $($($bounds)*)? > $Trait for $T {}
64        marker_impls! { $(#[$($meta)*])* $Trait for $($($rest)*)? }
65    },
66    ( $(#[$($meta:tt)*])* $Trait:ident for ) => {},
67
68    ( $(#[$($meta:tt)*])* unsafe $Trait:ident for $({$($bounds:tt)*})? $T:ty $(, $($rest:tt)*)? ) => {
69        $(#[$($meta)*])* unsafe impl< $($($bounds)*)? > $Trait for $T {}
70        marker_impls! { $(#[$($meta)*])* unsafe $Trait for $($($rest)*)? }
71    },
72    ( $(#[$($meta:tt)*])* unsafe $Trait:ident for ) => {},
73}

75/// Types that can be transferred across thread boundaries.
76///
77/// This trait is automatically implemented when the compiler determines it's
78/// appropriate.
79///
80/// An example of a non-`Send` type is the reference-counting pointer
81/// [`rc::Rc`][`Rc`]. If two threads attempt to clone [`Rc`]s that point to the same
82/// reference-counted value, they might try to update the reference count at the
83/// same time, which is [undefined behavior][ub] because [`Rc`] doesn't use atomic
84/// operations. Its cousin [`sync::Arc`][arc] does use atomic operations (incurring
85/// some overhead) and thus is `Send`.
86///
87/// See [the Nomicon](../../nomicon/send-and-sync.html) and the [`Sync`] trait for more details.
88///
89/// [`Rc`]: ../../std/rc/struct.Rc.html
90/// [arc]: ../../std/sync/struct.Arc.html
91/// [ub]: ../../reference/behavior-considered-undefined.html

93#[rustc_diagnostic_item = "Send"]
94#[diagnostic::on_unimplemented(
95    message = "`{Self}` cannot be sent between threads safely",
96    label = "`{Self}` cannot be sent between threads safely"
97)]
98pub unsafe auto trait Send {
99    // empty.
100}

113/// Types with a constant size known at compile time.
114///
115/// All type parameters have an implicit bound of `Sized`. The special syntax
116/// `?Sized` can be used to remove this bound if it's not appropriate.
117///
118/// ```
119/// # #![allow(dead_code)]
120/// struct Foo<T>(T);
121/// struct Bar<T: ?Sized>(T);
122///
123/// // struct FooUse(Foo<[i32]>); // error: Sized is not implemented for [i32]
124/// struct BarUse(Bar<[i32]>); // OK
125/// ```
126///
127/// The one exception is the implicit `Self` type of a trait. A trait does not
128/// have an implicit `Sized` bound as this is incompatible with [trait object]s
129/// where, by definition, the trait needs to work with all possible implementors,
130/// and thus could be any size.
131///
132/// Although Rust will let you bind `Sized` to a trait, you won't
133/// be able to use it to form a trait object later:
134///
135/// ```
136/// # #![allow(unused_variables)]
137/// trait Foo { }
138/// trait Bar: Sized { }
139///
140/// struct Impl;
141/// impl Foo for Impl { }
142/// impl Bar for Impl { }
143///
144/// let x: &dyn Foo = &Impl;    // OK
145/// // let y: &dyn Bar = &Impl; // error: the trait `Bar` cannot
146///                             // be made into an object
147/// ```
148///
149/// [trait object]: ../../book/ch17-02-trait-objects.html
150#[doc(alias = "?", alias = "?Sized")]
151#[stable(feature = "rust1", since = "1.0.0")]
152#[lang = "sized"]
153#[diagnostic::on_unimplemented(
154    message = "the size for values of type `{Self}` cannot be known at compilation time",
155    label = "doesn't have a size known at compile-time"
156)]
157#[fundamental] // for Default, for example, which requires that `[T]: !Default` be evaluatable
158#[rustc_specialization_trait]
159#[rustc_deny_explicit_impl]
160#[rustc_do_not_implement_via_object]
161#[rustc_coinductive]
162pub trait Sized {
163    // Empty.
164}

166/// Types that can be "unsized" to a dynamically-sized type.
167///
168/// For example, the sized array type `[i8; 2]` implements `Unsize<[i8]>` and
169/// `Unsize<dyn fmt::Debug>`.
170///
171/// All implementations of `Unsize` are provided automatically by the compiler.
172/// Those implementations are:
173///
174/// - Arrays `[T; N]` implement `Unsize<[T]>`.
175/// - A type implements `Unsize<dyn Trait + 'a>` if all of these conditions are met:
176///   - The type implements `Trait`.
177///   - `Trait` is dyn-compatible[^1].
178///   - The type is sized.
179///   - The type outlives `'a`.
180/// - Structs `Foo<..., T1, ..., Tn, ...>` implement `Unsize<Foo<..., U1, ..., Un, ...>>`
181/// where any number of (type and const) parameters may be changed if all of these conditions
182/// are met:
183///   - Only the last field of `Foo` has a type involving the parameters `T1`, ..., `Tn`.
184///   - All other parameters of the struct are equal.
185///   - `Field<T1, ..., Tn>: Unsize<Field<U1, ..., Un>>`, where `Field<...>` stands for the actual
186///     type of the struct's last field.
187///
188/// `Unsize` is used along with [`ops::CoerceUnsized`] to allow
189/// "user-defined" containers such as [`Rc`] to contain dynamically-sized
190/// types. See the [DST coercion RFC][RFC982] and [the nomicon entry on coercion][nomicon-coerce]
191/// for more details.
192///
193/// [`ops::CoerceUnsized`]: crate::ops::CoerceUnsized
194/// [`Rc`]: ../../std/rc/struct.Rc.html
195/// [RFC982]: https://github.com/rust-lang/rfcs/blob/master/text/0982-dst-coercion.md
196/// [nomicon-coerce]: ../../nomicon/coercions.html
197/// [^1]: Formerly known as *object safe*.
198#[unstable(feature = "unsize", issue = "18598")]
199#[lang = "unsize"]

201#[rustc_do_not_implement_via_object]

250/// Types whose values can be duplicated simply by copying bits.
251///
252/// By default, variable bindings have 'move semantics.' In other
253/// words:
254///
255/// ```
256/// #[derive(Debug)]
257/// struct Foo;
258///
259/// let x = Foo;
260///
261/// let y = x;
262///
263/// // `x` has moved into `y`, and so cannot be used
264///
265/// // println!("{x:?}"); // error: use of moved value
266/// ```
267///
268/// However, if a type implements `Copy`, it instead has 'copy semantics':
269///
270/// ```
271/// // We can derive a `Copy` implementation. `Clone` is also required, as it's
272/// // a supertrait of `Copy`.
273/// #[derive(Debug, Copy, Clone)]
274/// struct Foo;
275///
276/// let x = Foo;
277///
278/// let y = x;
279///
280/// // `y` is a copy of `x`
281///
282/// println!("{x:?}"); // A-OK!
283/// ```
284///
285/// It's important to note that in these two examples, the only difference is whether you
286/// are allowed to access `x` after the assignment. Under the hood, both a copy and a move
287/// can result in bits being copied in memory, although this is sometimes optimized away.
288///
289/// ## How can I implement `Copy`?
290///
291/// There are two ways to implement `Copy` on your type. The simplest is to use `derive`:
292///
293/// ```
294/// #[derive(Copy, Clone)]
295/// struct MyStruct;
296/// ```
297///
298/// You can also implement `Copy` and `Clone` manually:
299///
300/// ```
301/// struct MyStruct;
302///
303/// impl Copy for MyStruct { }
304///
305/// impl Clone for MyStruct {
306///     fn clone(&self) -> MyStruct {
307///         *self
308///     }
309/// }
310/// ```
311///
312/// There is a small difference between the two. The `derive` strategy will also place a `Copy`
313/// bound on type parameters:
314///
315/// ```
316/// #[derive(Clone)]
317/// struct MyStruct<T>(T);
318///
319/// impl<T: Copy> Copy for MyStruct<T> { }
320/// ```
321///
322/// This isn't always desired. For example, shared references (`&T`) can be copied regardless of
323/// whether `T` is `Copy`. Likewise, a generic struct containing markers such as [`PhantomData`]
324/// could potentially be duplicated with a bit-wise copy.
325///
326/// ## What's the difference between `Copy` and `Clone`?
327///
328/// Copies happen implicitly, for example as part of an assignment `y = x`. The behavior of
329/// `Copy` is not overloadable; it is always a simple bit-wise copy.
330///
331/// Cloning is an explicit action, `x.clone()`. The implementation of [`Clone`] can
332/// provide any type-specific behavior necessary to duplicate values safely. For example,
333/// the implementation of [`Clone`] for [`String`] needs to copy the pointed-to string
334/// buffer in the heap. A simple bitwise copy of [`String`] values would merely copy the
335/// pointer, leading to a double free down the line. For this reason, [`String`] is [`Clone`]
336/// but not `Copy`.
337///
338/// [`Clone`] is a supertrait of `Copy`, so everything which is `Copy` must also implement
339/// [`Clone`]. If a type is `Copy` then its [`Clone`] implementation only needs to return `*self`
340/// (see the example above).
341///
342/// ## When can my type be `Copy`?
343///
344/// A type can implement `Copy` if all of its components implement `Copy`. For example, this
345/// struct can be `Copy`:
346///
347/// ```
348/// # #[allow(dead_code)]
349/// #[derive(Copy, Clone)]
350/// struct Point {
351///    x: i32,
352///    y: i32,
353/// }
354/// ```
355///
356/// A struct can be `Copy`, and [`i32`] is `Copy`, therefore `Point` is eligible to be `Copy`.
357/// By contrast, consider
358///
359/// ```
360/// # #![allow(dead_code)]
361/// # struct Point;
362/// struct PointList {
363///     points: Vec<Point>,
364/// }
365/// ```
366///
367/// The struct `PointList` cannot implement `Copy`, because [`Vec<T>`] is not `Copy`. If we
368/// attempt to derive a `Copy` implementation, we'll get an error:
369///
370/// ```text
371/// the trait `Copy` cannot be implemented for this type; field `points` does not implement `Copy`
372/// ```
373///
374/// Shared references (`&T`) are also `Copy`, so a type can be `Copy`, even when it holds
375/// shared references of types `T` that are *not* `Copy`. Consider the following struct,
376/// which can implement `Copy`, because it only holds a *shared reference* to our non-`Copy`
377/// type `PointList` from above:
378///
379/// ```
380/// # #![allow(dead_code)]
381/// # struct PointList;
382/// #[derive(Copy, Clone)]
383/// struct PointListWrapper<'a> {
384///     point_list_ref: &'a PointList,
385/// }
386/// ```
387///
388/// ## When *can't* my type be `Copy`?
389///
390/// Some types can't be copied safely. For example, copying `&mut T` would create an aliased
391/// mutable reference. Copying [`String`] would duplicate responsibility for managing the
392/// [`String`]'s buffer, leading to a double free.
393///
394/// Generalizing the latter case, any type implementing [`Drop`] can't be `Copy`, because it's
395/// managing some resource besides its own [`size_of::<T>`] bytes.
396///
397/// If you try to implement `Copy` on a struct or enum containing non-`Copy` data, you will get
398/// the error [E0204].
399///
400/// [E0204]: ../../error_codes/E0204.html
401///
402/// ## When *should* my type be `Copy`?
403///
404/// Generally speaking, if your type _can_ implement `Copy`, it should. Keep in mind, though,
405/// that implementing `Copy` is part of the public API of your type. If the type might become
406/// non-`Copy` in the future, it could be prudent to omit the `Copy` implementation now, to
407/// avoid a breaking API change.
408///
409/// ## Additional implementors
410///
411/// In addition to the [implementors listed below][impls],
412/// the following types also implement `Copy`:
413///
414/// * Function item types (i.e., the distinct types defined for each function)
415/// * Function pointer types (e.g., `fn() -> i32`)
416/// * Closure types, if they capture no value from the environment
417///   or if all such captured values implement `Copy` themselves.
418///   Note that variables captured by shared reference always implement `Copy`
419///   (even if the referent doesn't),
420///   while variables captured by mutable reference never implement `Copy`.
421///
422/// [`Vec<T>`]: ../../std/vec/struct.Vec.html
423/// [`String`]: ../../std/string/struct.String.html
424/// [`size_of::<T>`]: size_of
425/// [impls]: #implementors
426#[stable(feature = "rust1", since = "1.0.0")]
427#[lang = "copy"]

431// We have this attribute here for now only because there are quite a few
432// existing specializations on `Copy` that already exist in the standard
433// library, and there's no way to safely have this behavior right now.
434#[rustc_unsafe_specialization_marker]

508/// Types for which it is safe to share references between threads.
509///
510/// This trait is automatically implemented when the compiler determines
511/// it's appropriate.
512///
513/// The precise definition is: a type `T` is [`Sync`] if and only if `&T` is
514/// [`Send`]. In other words, if there is no possibility of
515/// [undefined behavior][ub] (including data races) when passing
516/// `&T` references between threads.
517///
518/// As one would expect, primitive types like [`u8`] and [`f64`]
519/// are all [`Sync`], and so are simple aggregate types containing them,
520/// like tuples, structs and enums. More examples of basic [`Sync`]
521/// types include "immutable" types like `&T`, and those with simple
522/// inherited mutability, such as [`Box<T>`][box], [`Vec<T>`][vec] and
523/// most other collection types. (Generic parameters need to be [`Sync`]
524/// for their container to be [`Sync`].)
525///
526/// A somewhat surprising consequence of the definition is that `&mut T`
527/// is `Sync` (if `T` is `Sync`) even though it seems like that might
528/// provide unsynchronized mutation. The trick is that a mutable
529/// reference behind a shared reference (that is, `& &mut T`)
530/// becomes read-only, as if it were a `& &T`. Hence there is no risk
531/// of a data race.
532///
533/// A shorter overview of how [`Sync`] and [`Send`] relate to referencing:
534/// * `&T` is [`Send`] if and only if `T` is [`Sync`]
535/// * `&mut T` is [`Send`] if and only if `T` is [`Send`]
536/// * `&T` and `&mut T` are [`Sync`] if and only if `T` is [`Sync`]
537///
538/// Types that are not `Sync` are those that have "interior
539/// mutability" in a non-thread-safe form, such as [`Cell`][cell]
540/// and [`RefCell`][refcell]. These types allow for mutation of
541/// their contents even through an immutable, shared reference. For
542/// example the `set` method on [`Cell<T>`][cell] takes `&self`, so it requires
543/// only a shared reference [`&Cell<T>`][cell]. The method performs no
544/// synchronization, thus [`Cell`][cell] cannot be `Sync`.
545///
546/// Another example of a non-`Sync` type is the reference-counting
547/// pointer [`Rc`][rc]. Given any reference [`&Rc<T>`][rc], you can clone
548/// a new [`Rc<T>`][rc], modifying the reference counts in a non-atomic way.
549///
550/// For cases when one does need thread-safe interior mutability,
551/// Rust provides [atomic data types], as well as explicit locking via
552/// [`sync::Mutex`][mutex] and [`sync::RwLock`][rwlock]. These types
553/// ensure that any mutation cannot cause data races, hence the types
554/// are `Sync`. Likewise, [`sync::Arc`][arc] provides a thread-safe
555/// analogue of [`Rc`][rc].
556///
557/// Any types with interior mutability must also use the
558/// [`cell::UnsafeCell`][unsafecell] wrapper around the value(s) which
559/// can be mutated through a shared reference. Failing to doing this is
560/// [undefined behavior][ub]. For example, [`transmute`][transmute]-ing
561/// from `&T` to `&mut T` is invalid.
562///
563/// See [the Nomicon][nomicon-send-and-sync] for more details about `Sync`.
564///
565/// [box]: ../../std/boxed/struct.Box.html
566/// [vec]: ../../std/vec/struct.Vec.html
567/// [cell]: crate::cell::Cell
568/// [refcell]: crate::cell::RefCell
569/// [rc]: ../../std/rc/struct.Rc.html
570/// [arc]: ../../std/sync/struct.Arc.html
571/// [atomic data types]: crate::sync::atomic
572/// [mutex]: ../../std/sync/struct.Mutex.html
573/// [rwlock]: ../../std/sync/struct.RwLock.html
574/// [unsafecell]: crate::cell::UnsafeCell
575/// [ub]: ../../reference/behavior-considered-undefined.html
576/// [transmute]: crate::mem::transmute
577/// [nomicon-send-and-sync]: ../../nomicon/send-and-sync.html
578#[stable(feature = "rust1", since = "1.0.0")]
579#[rustc_diagnostic_item = "Sync"]
580#[lang = "sync"]
581#[rustc_on_unimplemented(
582    on(
583        _Self = "core::cell::once::OnceCell<T>",
584        note = "if you want to do aliasing and mutation between multiple threads, use `std::sync::OnceLock` instead"
585    ),
586    on(
587        _Self = "core::cell::Cell<u8>",

675/// Zero-sized type used to mark things that "act like" they own a `T`.
676///
677/// Adding a `PhantomData<T>` field to your type tells the compiler that your
678/// type acts as though it stores a value of type `T`, even though it doesn't
679/// really. This information is used when computing certain safety properties.
680///
681/// For a more in-depth explanation of how to use `PhantomData<T>`, please see
682/// [the Nomicon](../../nomicon/phantom-data.html).
683///
684/// # A ghastly note 
685///
686/// Though they both have scary names, `PhantomData` and 'phantom types' are
687/// related, but not identical. A phantom type parameter is simply a type
688/// parameter which is never used. In Rust, this often causes the compiler to
689/// complain, and the solution is to add a "dummy" use by way of `PhantomData`.
690///
691/// # Examples
692///
693/// ## Unused lifetime parameters
694///
695/// Perhaps the most common use case for `PhantomData` is a struct that has an
696/// unused lifetime parameter, typically as part of some unsafe code. For
697/// example, here is a struct `Slice` that has two pointers of type `*const T`,
698/// presumably pointing into an array somewhere:
699///
700/// ```compile_fail,E0392
701/// struct Slice<'a, T> {
702///     start: *const T,
703///     end: *const T,
704/// }
705/// ```
706///
707/// The intention is that the underlying data is only valid for the
708/// lifetime `'a`, so `Slice` should not outlive `'a`. However, this
709/// intent is not expressed in the code, since there are no uses of
710/// the lifetime `'a` and hence it is not clear what data it applies
711/// to. We can correct this by telling the compiler to act *as if* the
712/// `Slice` struct contained a reference `&'a T`:
713///
714/// ```
715/// use std::marker::PhantomData;
716///
717/// # #[allow(dead_code)]
718/// struct Slice<'a, T> {
719///     start: *const T,
720///     end: *const T,
721///     phantom: PhantomData<&'a T>,
722/// }
723/// ```
724///
725/// This also in turn infers the lifetime bound `T: 'a`, indicating
726/// that any references in `T` are valid over the lifetime `'a`.
727///
728/// When initializing a `Slice` you simply provide the value
729/// `PhantomData` for the field `phantom`:
730///
731/// ```
732/// # #![allow(dead_code)]
733/// # use std::marker::PhantomData;
734/// # struct Slice<'a, T> {
735/// #     start: *const T,
736/// #     end: *const T,
737/// #     phantom: PhantomData<&'a T>,
738/// # }
739/// fn borrow_vec<T>(vec: &Vec<T>) -> Slice<'_, T> {
740///     let ptr = vec.as_ptr();
741///     Slice {
742///         start: ptr,
743///         end: unsafe { ptr.add(vec.len()) },
744///         phantom: PhantomData,
745///     }
746/// }
747/// ```
748///
749/// ## Unused type parameters
750///
751/// It sometimes happens that you have unused type parameters which
752/// indicate what type of data a struct is "tied" to, even though that
753/// data is not actually found in the struct itself. Here is an
754/// example where this arises with [FFI]. The foreign interface uses
755/// handles of type `*mut ()` to refer to Rust values of different
756/// types. We track the Rust type using a phantom type parameter on
757/// the struct `ExternalResource` which wraps a handle.
758///
759/// [FFI]: ../../book/ch19-01-unsafe-rust.html#using-extern-functions-to-call-external-code
760///
761/// ```
762/// # #![allow(dead_code)]
763/// # trait ResType { }
764/// # struct ParamType;
765/// # mod foreign_lib {
766/// #     pub fn new(_: usize) -> *mut () { 42 as *mut () }
767/// #     pub fn do_stuff(_: *mut (), _: usize) {}
768/// # }
769/// # fn convert_params(_: ParamType) -> usize { 42 }
770/// use std::marker::PhantomData;
771///
772/// struct ExternalResource<R> {
773///    resource_handle: *mut (),
774///    resource_type: PhantomData<R>,
775/// }
776///
777/// impl<R: ResType> ExternalResource<R> {
778///     fn new() -> Self {
779///         let size_of_res = size_of::<R>();
780///         Self {
781///             resource_handle: foreign_lib::new(size_of_res),
782///             resource_type: PhantomData,
783///         }
784///     }
785///
786///     fn do_stuff(&self, param: ParamType) {
787///         let foreign_params = convert_params(param);
788///         foreign_lib::do_stuff(self.resource_handle, foreign_params);
789///     }
790/// }
791/// ```
792///
793/// ## Ownership and the drop check
794///
795/// The exact interaction of `PhantomData` with drop check **may change in the future**.
796///
797/// Currently, adding a field of type `PhantomData<T>` indicates that your type *owns* data of type
798/// `T` in very rare circumstances. This in turn has effects on the Rust compiler's [drop check]
799/// analysis. For the exact rules, see the [drop check] documentation.
800///
801/// ## Layout
802///
803/// For all `T`, the following are guaranteed:
804/// * `size_of::<PhantomData<T>>() == 0`
805/// * `align_of::<PhantomData<T>>() == 1`
806///
807/// [drop check]: Drop#drop-check
808#[lang = "phantom_data"]
809#[stable(feature = "rust1", since = "1.0.0")]
810pub struct PhantomData<T: ?Sized>;

954/// Types that do not require any pinning guarantees.
955///
956/// For information on what "pinning" is, see the [`pin` module] documentation.
957///
958/// Implementing the `Unpin` trait for `T` expresses the fact that `T` is pinning-agnostic:
959/// it shall not expose nor rely on any pinning guarantees. This, in turn, means that a
960/// `Pin`-wrapped pointer to such a type can feature a *fully unrestricted* API.
961/// In other words, if `T: Unpin`, a value of type `T` will *not* be bound by the invariants
962/// which pinning otherwise offers, even when "pinned" by a [`Pin<Ptr>`] pointing at it.
963/// When a value of type `T` is pointed at by a [`Pin<Ptr>`], [`Pin`] will not restrict access
964/// to the pointee value like it normally would, thus allowing the user to do anything that they
965/// normally could with a non-[`Pin`]-wrapped `Ptr` to that value.
966///
967/// The idea of this trait is to alleviate the reduced ergonomics of APIs that require the use
968/// of [`Pin`] for soundness for some types, but which also want to be used by other types that
969/// don't care about pinning. The prime example of such an API is [`Future::poll`]. There are many
970/// [`Future`] types that don't care about pinning. These futures can implement `Unpin` and
971/// therefore get around the pinning related restrictions in the API, while still allowing the
972/// subset of [`Future`]s which *do* require pinning to be implemented soundly.
973///
974/// For more discussion on the consequences of [`Unpin`] within the wider scope of the pinning
975/// system, see the [section about `Unpin`] in the [`pin` module].
976///
977/// `Unpin` has no consequence at all for non-pinned data. In particular, [`mem::replace`] happily
978/// moves `!Unpin` data, which would be immovable when pinned ([`mem::replace`] works for any
979/// `&mut T`, not just when `T: Unpin`).
980///
981/// *However*, you cannot use [`mem::replace`] on `!Unpin` data which is *pinned* by being wrapped
982/// inside a [`Pin<Ptr>`] pointing at it. This is because you cannot (safely) use a
983/// [`Pin<Ptr>`] to get a `&mut T` to its pointee value, which you would need to call
984/// [`mem::replace`], and *that* is what makes this system work.
985///
986/// So this, for example, can only be done on types implementing `Unpin`:
987///
988/// ```rust
989/// # #![allow(unused_must_use)]
990/// use std::mem;
991/// use std::pin::Pin;
992///
993/// let mut string = "this".to_string();
994/// let mut pinned_string = Pin::new(&mut string);
995///
996/// // We need a mutable reference to call `mem::replace`.
997/// // We can obtain such a reference by (implicitly) invoking `Pin::deref_mut`,
998/// // but that is only possible because `String` implements `Unpin`.
999/// mem::replace(&mut *pinned_string, "other".to_string());
1000/// ```
1001///
1002/// This trait is automatically implemented for almost every type. The compiler is free
1003/// to take the conservative stance of marking types as [`Unpin`] so long as all of the types that
1004/// compose its fields are also [`Unpin`]. This is because if a type implements [`Unpin`], then it
1005/// is unsound for that type's implementation to rely on pinning-related guarantees for soundness,
1006/// *even* when viewed through a "pinning" pointer! It is the responsibility of the implementor of
1007/// a type that relies upon pinning for soundness to ensure that type is *not* marked as [`Unpin`]
1008/// by adding [`PhantomPinned`] field. For more details, see the [`pin` module] docs.
1009///
1010/// [`mem::replace`]: crate::mem::replace "mem replace"
1011/// [`Future`]: crate::future::Future "Future"
1012/// [`Future::poll`]: crate::future::Future::poll "Future poll"
1013/// [`Pin`]: crate::pin::Pin "Pin"
1014/// [`Pin<Ptr>`]: crate::pin::Pin "Pin"
1015/// [`pin` module]: crate::pin "pin module"
1016/// [section about `Unpin`]: crate::pin#unpin "pin module docs about unpin"
1017/// [`unsafe`]: ../../std/keyword.unsafe.html "keyword unsafe"
1018#[stable(feature = "pin", since = "1.33.0")]
1019#[diagnostic::on_unimplemented(
1020    note = "consider using the `pin!` macro\nconsider using `Box::pin` if you need to access the pinned value outside of the current scope",
1021    message = "`{Self}` cannot be unpinned"
1022)]
1023#[lang = "unpin"]

1025pub auto trait Unpin {}

1231/// Derive macro that makes a smart pointer usable with trait objects.
1232///
1233/// # What this macro does
1234///
1235/// This macro is intended to be used with user-defined pointer types, and makes it possible to
1236/// perform coercions on the pointee of the user-defined pointer. There are two aspects to this:
1237///
1238/// ## Unsizing coercions of the pointee
1239///
1240/// By using the macro, the following example will compile:
1241/// ```
1242/// #![feature(derive_coerce_pointee)]
1243/// use std::marker::CoercePointee;
1244/// use std::ops::Deref;
1245///
1246/// #[derive(CoercePointee)]
1247/// #[repr(transparent)]
1248/// struct MySmartPointer<T: ?Sized>(Box<T>);
1249///
1250/// impl<T: ?Sized> Deref for MySmartPointer<T> {
1251///     type Target = T;
1252///     fn deref(&self) -> &T {
1253///         &self.0
1254///     }
1255/// }
1256///
1257/// trait MyTrait {}
1258///
1259/// impl MyTrait for i32 {}
1260///
1261/// fn main() {
1262///     let ptr: MySmartPointer<i32> = MySmartPointer(Box::new(4));
1263///
1264///     // This coercion would be an error without the derive.
1265///     let ptr: MySmartPointer<dyn MyTrait> = ptr;
1266/// }
1267/// ```
1268/// Without the `#[derive(CoercePointee)]` macro, this example would fail with the following error:
1269/// ```text
1270/// error[E0308]: mismatched types
1271///   --> src/main.rs:11:44
1272///    |
1273/// 11 |     let ptr: MySmartPointer<dyn MyTrait> = ptr;
1274///    |              ---------------------------   ^^^ expected `MySmartPointer<dyn MyTrait>`, found `MySmartPointer<i32>`
1275///    |              |
1276///    |              expected due to this
1277///    |
1278///    = note: expected struct `MySmartPointer<dyn MyTrait>`
1279///               found struct `MySmartPointer<i32>`
1280///    = help: `i32` implements `MyTrait` so you could box the found value and coerce it to the trait object `Box<dyn MyTrait>`, you will have to change the expected type as well
1281/// ```
1282///
1283/// ## Dyn compatibility
1284///
1285/// This macro allows you to dispatch on the user-defined pointer type. That is, traits using the
1286/// type as a receiver are dyn-compatible. For example, this compiles:
1287///
1288/// ```
1289/// #![feature(arbitrary_self_types, derive_coerce_pointee)]
1290/// use std::marker::CoercePointee;
1291/// use std::ops::Deref;
1292///
1293/// #[derive(CoercePointee)]
1294/// #[repr(transparent)]
1295/// struct MySmartPointer<T: ?Sized>(Box<T>);
1296///
1297/// impl<T: ?Sized> Deref for MySmartPointer<T> {
1298///     type Target = T;
1299///     fn deref(&self) -> &T {
1300///         &self.0
1301///     }
1302/// }
1303///
1304/// // You can always define this trait. (as long as you have #![feature(arbitrary_self_types)])
1305/// trait MyTrait {
1306///     fn func(self: MySmartPointer<Self>);
1307/// }
1308///
1309/// // But using `dyn MyTrait` requires #[derive(CoercePointee)].
1310/// fn call_func(value: MySmartPointer<dyn MyTrait>) {
1311///     value.func();
1312/// }
1313/// ```
1314/// If you remove the `#[derive(CoercePointee)]` annotation from the struct, then the above example
1315/// will fail with this error message:
1316/// ```text
1317/// error[E0038]: the trait `MyTrait` is not dyn compatible
1318///   --> src/lib.rs:21:36
1319///    |
1320/// 17 |     fn func(self: MySmartPointer<Self>);
1321///    |                   -------------------- help: consider changing method `func`'s `self` parameter to be `&self`: `&Self`
1322/// ...
1323/// 21 | fn call_func(value: MySmartPointer<dyn MyTrait>) {
1324///    |                                    ^^^^^^^^^^^ `MyTrait` is not dyn compatible
1325///    |
1326/// note: for a trait to be dyn compatible it needs to allow building a vtable
1327///       for more information, visit <https://doc.rust-lang.org/reference/items/traits.html#object-safety>
1328///   --> src/lib.rs:17:19
1329///    |
1330/// 16 | trait MyTrait {
1331///    |       ------- this trait is not dyn compatible...
1332/// 17 |     fn func(self: MySmartPointer<Self>);
1333///    |                   ^^^^^^^^^^^^^^^^^^^^ ...because method `func`'s `self` parameter cannot be dispatched on
1334/// ```
1335///
1336/// # Requirements for using the macro
1337///
1338/// This macro can only be used if:
1339/// * The type is a `#[repr(transparent)]` struct.
1340/// * The type of its non-zero-sized field must either be a standard library pointer type
1341///   (reference, raw pointer, `NonNull`, `Box`, `Rc`, `Arc`, etc.) or another user-defined type
1342///   also using the `#[derive(CoercePointee)]` macro.
1343/// * Zero-sized fields must not mention any generic parameters unless the zero-sized field has
1344///   type [`PhantomData`].
1345///
1346/// ## Multiple type parameters
1347///
1348/// If the type has multiple type parameters, then you must explicitly specify which one should be
1349/// used for dynamic dispatch. For example:
1350/// ```
1351/// # #![feature(derive_coerce_pointee)]
1352/// # use std::marker::{CoercePointee, PhantomData};
1353/// #[derive(CoercePointee)]
1354/// #[repr(transparent)]
1355/// struct MySmartPointer<#[pointee] T: ?Sized, U> {
1356///     ptr: Box<T>,
1357///     _phantom: PhantomData<U>,
1358/// }
1359/// ```
1360/// Specifying `#[pointee]` when the struct has only one type parameter is allowed, but not required.
1361///
1362/// # Examples
1363///
1364/// A custom implementation of the `Rc` type:
1365/// ```
1366/// #![feature(derive_coerce_pointee)]
1367/// use std::marker::CoercePointee;
1368/// use std::ops::Deref;
1369/// use std::ptr::NonNull;
1370///
1371/// #[derive(CoercePointee)]
1372/// #[repr(transparent)]
1373/// pub struct Rc<T: ?Sized> {
1374///     inner: NonNull<RcInner<T>>,
1375/// }
1376///
1377/// struct RcInner<T: ?Sized> {
1378///     refcount: usize,
1379///     value: T,
1380/// }
1381///
1382/// impl<T: ?Sized> Deref for Rc<T> {
1383///     type Target = T;
1384///     fn deref(&self) -> &T {
1385///         let ptr = self.inner.as_ptr();
1386///         unsafe { &(*ptr).value }
1387///     }
1388/// }
1389///
1390/// impl<T> Rc<T> {
1391///     pub fn new(value: T) -> Self {
1392///         let inner = Box::new(RcInner {
1393///             refcount: 1,
1394///             value,
1395///         });
1396///         Self {
1397///             inner: NonNull::from(Box::leak(inner)),
1398///         }
1399///     }
1400/// }
1401///
1402/// impl<T: ?Sized> Clone for Rc<T> {
1403///     fn clone(&self) -> Self {
1404///         // A real implementation would handle overflow here.
1405///         unsafe { (*self.inner.as_ptr()).refcount += 1 };
1406///         Self { inner: self.inner }
1407///     }
1408/// }
1409///
1410/// impl<T: ?Sized> Drop for Rc<T> {
1411///     fn drop(&mut self) {
1412///         let ptr = self.inner.as_ptr();
1413///         unsafe { (*ptr).refcount -= 1 };
1414///         if unsafe { (*ptr).refcount } == 0 {
1415///             drop(unsafe { Box::from_raw(ptr) });
1416///         }
1417///     }
1418/// }
1419/// ```
1420#[rustc_builtin_macro(CoercePointee, attributes(pointee))]
1421#[allow_internal_unstable(dispatch_from_dyn, coerce_unsized, unsize, coerce_pointee_validated)]
1422#[rustc_diagnostic_item = "CoercePointee"]
1423#[unstable(feature = "derive_coerce_pointee", issue = "123430")]
1424#[cfg(not(feature = "ferrocene_certified"))]
1425pub macro CoercePointee($item:item) {
1426    /* compiler built-in */
1427}
