1//! Panic support for core
2//!
3//! In core, panicking is always done with a message, resulting in a `core::panic::PanicInfo`
4//! containing a `fmt::Arguments`. In std, however, panicking can be done with panic_any, which
5//! throws a `Box<dyn Any>` containing any type of value. Because of this,
6//! `std::panic::PanicHookInfo` is a different type, which contains a `&dyn Any` instead of a
7//! `fmt::Arguments`. std's panic handler will convert the `fmt::Arguments` to a `&dyn Any`
8//! containing either a `&'static str` or `String` containing the formatted message.
9//!
10//! The core library cannot define any panic handler, but it can invoke it.
11//! This means that the functions inside of core are allowed to panic, but to be
12//! useful an upstream crate must define panicking for core to use. The current
13//! interface for panicking is:
14//!
15//! ```
16//! fn panic_impl(pi: &core::panic::PanicInfo<'_>) -> !
17//! # { loop {} }
18//! ```
19//!
20//! This module contains a few other panicking functions, but these are just the
21//! necessary lang items for the compiler. All panics are funneled through this
22//! one function. The actual symbol is declared through the `#[panic_handler]` attribute.
23

44/// The entry point for panicking with a formatted message.
45///
46/// This is designed to reduce the amount of code required at the call
47/// site as much as possible (so that `panic!()` has as low an impact
48/// on (e.g.) the inlining of other functions as possible), by moving
49/// the actual formatting into this shared place.
50// If panic_immediate_abort, inline the abort call,
51// otherwise avoid inlining because of it is cold path.
52#[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold)]
53#[cfg_attr(feature = "panic_immediate_abort", inline)]
54#[track_caller]
55#[lang = "panic_fmt"] // needed for const-evaluated panics
56#[rustc_do_not_const_check] // hooked by const-eval
57#[rustc_const_stable_indirect] // must follow stable const rules since it is exposed to stable
58#[cfg(not(feature = "ferrocene_certified"))]
59pub const fn panic_fmt(fmt: fmt::Arguments<'_>) -> ! {
60    if cfg!(feature = "panic_immediate_abort") {
61        super::intrinsics::abort()
62    }
63
64    // NOTE This function never crosses the FFI boundary; it's a Rust-to-Rust call
65    // that gets resolved to the `#[panic_handler]` function.
66    unsafe extern "Rust" {
67        #[lang = "panic_impl"]
68        fn panic_impl(pi: &PanicInfo<'_>) -> !;
69    }
70
71    let pi = PanicInfo::new(
72        &fmt,
73        Location::caller(),
74        /* can_unwind */ true,
75        /* force_no_backtrace */ false,
76    );
77
78    // SAFETY: `panic_impl` is defined in safe Rust code and thus is safe to call.
79    unsafe { panic_impl(&pi) }
80}

93/// Like `panic_fmt`, but for non-unwinding panics.
94///
95/// Has to be a separate function so that it can carry the `rustc_nounwind` attribute.
96#[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold)]
97#[cfg_attr(feature = "panic_immediate_abort", inline)]
98#[track_caller]
99// This attribute has the key side-effect that if the panic handler ignores `can_unwind`
100// and unwinds anyway, we will hit the "unwinding out of nounwind function" guard,
101// which causes a "panic in a function that cannot unwind".
102#[rustc_nounwind]
103#[rustc_const_stable_indirect] // must follow stable const rules since it is exposed to stable
104#[rustc_allow_const_fn_unstable(const_eval_select)]
105#[cfg(not(feature = "ferrocene_certified"))]
106pub const fn panic_nounwind_fmt(fmt: fmt::Arguments<'_>, force_no_backtrace: bool) -> ! {
107    const_eval_select!(
108        @capture { fmt: fmt::Arguments<'_>, force_no_backtrace: bool } -> !:
109        if const #[track_caller] {
110            // We don't unwind anyway at compile-time so we can call the regular `panic_fmt`.
111            panic_fmt(fmt)
112        } else #[track_caller] {
113            if cfg!(feature = "panic_immediate_abort") {
114                super::intrinsics::abort()
115            }
116
117            // NOTE This function never crosses the FFI boundary; it's a Rust-to-Rust call
118            // that gets resolved to the `#[panic_handler]` function.
119            unsafe extern "Rust" {
120                #[lang = "panic_impl"]
121                fn panic_impl(pi: &PanicInfo<'_>) -> !;
122            }
123
124            // PanicInfo with the `can_unwind` flag set to false forces an abort.
125            let pi = PanicInfo::new(
126                &fmt,
127                Location::caller(),
128                /* can_unwind */ false,
129                force_no_backtrace,
130            );
131
132            // SAFETY: `panic_impl` is defined in safe Rust code and thus is safe to call.
133            unsafe { panic_impl(&pi) }
134        }
135    )
136}

163/// The underlying implementation of core's `panic!` macro when no formatting is used.
164// Never inline unless panic_immediate_abort to avoid code
165// bloat at the call sites as much as possible.
166#[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold)]
167#[cfg_attr(feature = "panic_immediate_abort", inline)]
168#[track_caller]
169#[rustc_const_stable_indirect] // must follow stable const rules since it is exposed to stable
170#[lang = "panic"] // used by lints and miri for panics
171pub const fn panic(expr: &'static str) -> ! {
172    // Use Arguments::new_const instead of format_args!("{expr}") to potentially
173    // reduce size overhead. The format_args! macro uses str's Display trait to
174    // write expr, which calls Formatter::pad, which must accommodate string
175    // truncation and padding (even though none is used here). Using
176    // Arguments::new_const may allow the compiler to omit Formatter::pad from the
177    // output binary, saving up to a few kilobytes.
178    // However, this optimization only works for `'static` strings: `new_const` also makes this
179    // message return `Some` from `Arguments::as_str`, which means it can become part of the panic
180    // payload without any allocation or copying. Shorter-lived strings would become invalid as
181    // stack frames get popped during unwinding, and couldn't be directly referenced from the
182    // payload.
183    #[cfg(not(feature = "ferrocene_certified"))]
184    panic_fmt(fmt::Arguments::new_const(&[expr]));
185    #[cfg(feature = "ferrocene_certified")]
186    panic_fmt(expr)
187}

189// We generate functions for usage by compiler-generated assertions.
190//
191// Placing these functions in libcore means that all Rust programs can generate a jump into this
192// code rather than expanding to panic("...") above, which adds extra bloat to call sites (for the
193// constant string argument's pointer and length).
194//
195// This is especially important when this code is called often (e.g., with -Coverflow-checks) for
196// reducing binary size impact.
197#[cfg(not(feature = "ferrocene_certified"))]
198macro_rules! panic_const {
199    ($($lang:ident = $message:expr,)+) => {
200        $(
201            /// This is a panic called with a message that's a result of a MIR-produced Assert.
202            //
203            // never inline unless panic_immediate_abort to avoid code
204            // bloat at the call sites as much as possible
205            #[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold)]
206            #[cfg_attr(feature = "panic_immediate_abort", inline)]
207            #[track_caller]
208            #[rustc_const_stable_indirect] // must follow stable const rules since it is exposed to stable
209            #[lang = stringify!($lang)]
210            pub const fn $lang() -> ! {
211                // Use Arguments::new_const instead of format_args!("{expr}") to potentially
212                // reduce size overhead. The format_args! macro uses str's Display trait to
213                // write expr, which calls Formatter::pad, which must accommodate string
214                // truncation and padding (even though none is used here). Using
215                // Arguments::new_const may allow the compiler to omit Formatter::pad from the
216                // output binary, saving up to a few kilobytes.
217                panic_fmt(fmt::Arguments::new_const(&[$message]));
218            }
219        )+
220    }
221}

405/// This function is used instead of panic_fmt in const eval.
406#[lang = "const_panic_fmt"] // needed by const-eval machine to replace calls to `panic_fmt` lang item
407#[rustc_const_stable_indirect] // must follow stable const rules since it is exposed to stable
408#[cfg(not(feature = "ferrocene_certified"))]
409pub const fn const_panic_fmt(fmt: fmt::Arguments<'_>) -> ! {
410    if let Some(msg) = fmt.as_str() {
411        // The panic_display function is hooked by const eval.
412        panic_display(&msg);
413    } else {
414        // SAFETY: This is only evaluated at compile time, which reliably
415        // handles this UB (in case this branch turns out to be reachable
416        // somehow).
417        unsafe { crate::hint::unreachable_unchecked() };
418    }
419}

436pub fn assert_failed<T, U>(
437    kind: AssertKind,
438    left: &T,
439    right: &U,
440    args: Option<fmt::Arguments<'_>>,
441) -> !
442where
443    T: fmt::Debug + ?Sized,
444    U: fmt::Debug + ?Sized,

446    assert_failed_inner(kind, &left, &right, args)

449/// Internal function for `assert_match!`
450#[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold, optimize(size))]
451#[cfg_attr(feature = "panic_immediate_abort", inline)]
452#[track_caller]
453#[doc(hidden)]
454#[cfg(not(feature = "ferrocene_certified"))]
455pub fn assert_matches_failed<T: fmt::Debug + ?Sized>(
456    left: &T,
457    right: &str,
458    args: Option<fmt::Arguments<'_>>,
459) -> ! {
460    // The pattern is a string so it can be displayed directly.
461    struct Pattern<'a>(&'a str);
462    impl fmt::Debug for Pattern<'_> {
463        fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
464            f.write_str(self.0)
465        }
466    }
467    assert_failed_inner(AssertKind::Match, &left, &Pattern(right), args);
468}

470/// Non-generic version of the above functions, to avoid code bloat.
471#[cfg_attr(not(feature = "panic_immediate_abort"), inline(never), cold, optimize(size))]
472#[cfg_attr(feature = "panic_immediate_abort", inline)]
473#[track_caller]
474#[cfg(not(feature = "ferrocene_certified"))]
475fn assert_failed_inner(
476    kind: AssertKind,
477    left: &dyn fmt::Debug,
478    right: &dyn fmt::Debug,
479    args: Option<fmt::Arguments<'_>>,
480) -> ! {
481    let op = match kind {
482        AssertKind::Eq => "==",
483        AssertKind::Ne => "!=",
484        AssertKind::Match => "matches",
485    };
486
487    match args {
488        Some(args) => panic!(
489            r#"assertion `left {op} right` failed: {args}
490  left: {left:?}
491 right: {right:?}"#
492        ),
493        None => panic!(
494            r#"assertion `left {op} right` failed
495  left: {left:?}
496 right: {right:?}"#
497        ),
498    }
499}
