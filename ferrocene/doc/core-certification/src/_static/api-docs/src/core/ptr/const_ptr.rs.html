15    /// Returns `true` if the pointer is null.
16    ///
17    /// Note that unsized types have many possible null pointers, as only the
18    /// raw data pointer is considered, not their length, vtable, etc.
19    /// Therefore, two pointers that are null may still not compare equal to
20    /// each other.
21    ///
22    /// # Panics during const evaluation
23    ///
24    /// If this method is used during const evaluation, and `self` is a pointer
25    /// that is offset beyond the bounds of the memory it initially pointed to,
26    /// then there might not be enough information to determine whether the
27    /// pointer is null. This is because the absolute address in memory is not
28    /// known at compile time. If the nullness of the pointer cannot be
29    /// determined, this method will panic.
30    ///
31    /// In-bounds pointers are never null, so the method will never panic for
32    /// such pointers.
33    ///
34    /// # Examples
35    ///
36    /// ```
37    /// let s: &str = "Follow the rabbit";
38    /// let ptr: *const u8 = s.as_ptr();
39    /// assert!(!ptr.is_null());
40    /// ```
41    #[stable(feature = "rust1", since = "1.0.0")]
42    #[rustc_const_stable(feature = "const_ptr_is_null", since = "1.84.0")]
43    #[rustc_diagnostic_item = "ptr_const_is_null"]
44    #[inline]
45    #[rustc_allow_const_fn_unstable(const_eval_select)]
46    pub const fn is_null(self) -> bool {
47        // Compare via a cast to a thin pointer, so fat pointers are only
48        // considering their "data" part for null-ness.
49        let ptr = self as *const u8;
50        const_eval_select!(
51            @capture { ptr: *const u8 } -> bool:
52            // This use of `const_raw_ptr_comparison` has been explicitly blessed by t-lang.
53            if const #[rustc_allow_const_fn_unstable(const_raw_ptr_comparison)] {
54                match (ptr).guaranteed_eq(null_mut()) {
55                    Some(res) => res,
56                    // To remain maximally convervative, we stop execution when we don't
57                    // know whether the pointer is null or not.
58                    // We can *not* return `false` here, that would be unsound in `NonNull::new`!
59                    None => panic!("null-ness of this pointer cannot be determined in const context"),
60                }
61            } else {
62                ptr.addr() == 0
63            }
64        )
65    }

76    /// Uses the address value in a new pointer of another type.
77    ///
78    /// This operation will ignore the address part of its `meta` operand and discard existing
79    /// metadata of `self`. For pointers to a sized types (thin pointers), this has the same effect
80    /// as a simple cast. For pointers to an unsized type (fat pointers) this recombines the address
81    /// with new metadata such as slice lengths or `dyn`-vtable.
82    ///
83    /// The resulting pointer will have provenance of `self`. This operation is semantically the
84    /// same as creating a new pointer with the data pointer value of `self` but the metadata of
85    /// `meta`, being fat or thin depending on the `meta` operand.
86    ///
87    /// # Examples
88    ///
89    /// This function is primarily useful for enabling pointer arithmetic on potentially fat
90    /// pointers. The pointer is cast to a sized pointee to utilize offset operations and then
91    /// recombined with its own original metadata.
92    ///
93    /// ```
94    /// #![feature(set_ptr_value)]
95    /// # use core::fmt::Debug;
96    /// let arr: [i32; 3] = [1, 2, 3];
97    /// let mut ptr = arr.as_ptr() as *const dyn Debug;
98    /// let thin = ptr as *const u8;
99    /// unsafe {
100    ///     ptr = thin.add(8).with_metadata_of(ptr);
101    ///     # assert_eq!(*(ptr as *const i32), 3);
102    ///     println!("{:?}", &*ptr); // will print "3"
103    /// }
104    /// ```
105    ///
106    /// # *Incorrect* usage
107    ///
108    /// The provenance from pointers is *not* combined. The result must only be used to refer to the
109    /// address allowed by `self`.
110    ///
111    /// ```rust,no_run
112    /// #![feature(set_ptr_value)]
113    /// let x = 0u32;
114    /// let y = 1u32;
115    ///
116    /// let x = (&x) as *const u32;
117    /// let y = (&y) as *const u32;
118    ///
119    /// let offset = (x as usize - y as usize) / 4;
120    /// let bad = x.wrapping_add(offset).with_metadata_of(y);
121    ///
122    /// // This dereference is UB. The pointer only has provenance for `x` but points to `y`.
123    /// println!("{:?}", unsafe { &*bad });
124    /// ```
125    #[unstable(feature = "set_ptr_value", issue = "75091")]
126    #[must_use = "returns a new pointer rather than modifying its argument"]
127    #[inline]
128    #[cfg(not(feature = "ferrocene_certified"))]
129    pub const fn with_metadata_of<U>(self, meta: *const U) -> *const U
130    where
131        U: ?Sized,
132    {
133        from_raw_parts::<U>(self as *const (), metadata(meta))
134    }

149    /// Gets the "address" portion of the pointer.
150    ///
151    /// This is similar to `self as usize`, except that the [provenance][crate::ptr#provenance] of
152    /// the pointer is discarded and not [exposed][crate::ptr#exposed-provenance]. This means that
153    /// casting the returned address back to a pointer yields a [pointer without
154    /// provenance][without_provenance], which is undefined behavior to dereference. To properly
155    /// restore the lost information and obtain a dereferenceable pointer, use
156    /// [`with_addr`][pointer::with_addr] or [`map_addr`][pointer::map_addr].
157    ///
158    /// If using those APIs is not possible because there is no way to preserve a pointer with the
159    /// required provenance, then Strict Provenance might not be for you. Use pointer-integer casts
160    /// or [`expose_provenance`][pointer::expose_provenance] and [`with_exposed_provenance`][with_exposed_provenance]
161    /// instead. However, note that this makes your code less portable and less amenable to tools
162    /// that check for compliance with the Rust memory model.
163    ///
164    /// On most platforms this will produce a value with the same bytes as the original
165    /// pointer, because all the bytes are dedicated to describing the address.
166    /// Platforms which need to store additional information in the pointer may
167    /// perform a change of representation to produce a value containing only the address
168    /// portion of the pointer. What that means is up to the platform to define.
169    ///
170    /// This is a [Strict Provenance][crate::ptr#strict-provenance] API.
171    #[must_use]
172    #[inline(always)]
173    #[stable(feature = "strict_provenance", since = "1.84.0")]
174    pub fn addr(self) -> usize {
175        // A pointer-to-integer transmute currently has exactly the right semantics: it returns the
176        // address without exposing the provenance. Note that this is *not* a stable guarantee about
177        // transmute semantics, it relies on sysroot crates having special status.
178        // SAFETY: Pointer-to-integer transmutes are valid (if you are okay with losing the
179        // provenance).
180        unsafe { mem::transmute(self.cast::<()>()) }
181    }

183    /// Exposes the ["provenance"][crate::ptr#provenance] part of the pointer for future use in
184    /// [`with_exposed_provenance`] and returns the "address" portion.
185    ///
186    /// This is equivalent to `self as usize`, which semantically discards provenance information.
187    /// Furthermore, this (like the `as` cast) has the implicit side-effect of marking the
188    /// provenance as 'exposed', so on platforms that support it you can later call
189    /// [`with_exposed_provenance`] to reconstitute the original pointer including its provenance.
190    ///
191    /// Due to its inherent ambiguity, [`with_exposed_provenance`] may not be supported by tools
192    /// that help you to stay conformant with the Rust memory model. It is recommended to use
193    /// [Strict Provenance][crate::ptr#strict-provenance] APIs such as [`with_addr`][pointer::with_addr]
194    /// wherever possible, in which case [`addr`][pointer::addr] should be used instead of `expose_provenance`.
195    ///
196    /// On most platforms this will produce a value with the same bytes as the original pointer,
197    /// because all the bytes are dedicated to describing the address. Platforms which need to store
198    /// additional information in the pointer may not support this operation, since the 'expose'
199    /// side-effect which is required for [`with_exposed_provenance`] to work is typically not
200    /// available.
201    ///
202    /// This is an [Exposed Provenance][crate::ptr#exposed-provenance] API.
203    ///
204    /// [`with_exposed_provenance`]: with_exposed_provenance
205    #[inline(always)]
206    #[stable(feature = "exposed_provenance", since = "1.84.0")]
207    #[cfg(not(feature = "ferrocene_certified"))]
208    pub fn expose_provenance(self) -> usize {
209        self.cast::<()>() as usize
210    }

212    /// Creates a new pointer with the given address and the [provenance][crate::ptr#provenance] of
213    /// `self`.
214    ///
215    /// This is similar to a `addr as *const T` cast, but copies
216    /// the *provenance* of `self` to the new pointer.
217    /// This avoids the inherent ambiguity of the unary cast.
218    ///
219    /// This is equivalent to using [`wrapping_offset`][pointer::wrapping_offset] to offset
220    /// `self` to the given address, and therefore has all the same capabilities and restrictions.
221    ///
222    /// This is a [Strict Provenance][crate::ptr#strict-provenance] API.
223    #[must_use]
224    #[inline]
225    #[stable(feature = "strict_provenance", since = "1.84.0")]
226    #[cfg(not(feature = "ferrocene_certified"))]
227    pub fn with_addr(self, addr: usize) -> Self {
228        // This should probably be an intrinsic to avoid doing any sort of arithmetic, but
229        // meanwhile, we can implement it with `wrapping_offset`, which preserves the pointer's
230        // provenance.
231        let self_addr = self.addr() as isize;
232        let dest_addr = addr as isize;
233        let offset = dest_addr.wrapping_sub(self_addr);
234        self.wrapping_byte_offset(offset)
235    }

390    /// Adds a signed offset to a pointer.
391    ///
392    /// `count` is in units of T; e.g., a `count` of 3 represents a pointer
393    /// offset of `3 * size_of::<T>()` bytes.
394    ///
395    /// # Safety
396    ///
397    /// If any of the following conditions are violated, the result is Undefined Behavior:
398    ///
399    /// * The offset in bytes, `count * size_of::<T>()`, computed on mathematical integers (without
400    ///   "wrapping around"), must fit in an `isize`.
401    ///
402    /// * If the computed offset is non-zero, then `self` must be [derived from][crate::ptr#provenance] a pointer to some
403    ///   [allocated object], and the entire memory range between `self` and the result must be in
404    ///   bounds of that allocated object. In particular, this range must not "wrap around" the edge
405    ///   of the address space. Note that "range" here refers to a half-open range as usual in Rust,
406    ///   i.e., `self..result` for non-negative offsets and `result..self` for negative offsets.
407    ///
408    /// Allocated objects can never be larger than `isize::MAX` bytes, so if the computed offset
409    /// stays in bounds of the allocated object, it is guaranteed to satisfy the first requirement.
410    /// This implies, for instance, that `vec.as_ptr().add(vec.len())` (for `vec: Vec<T>`) is always
411    /// safe.
412    ///
413    /// Consider using [`wrapping_offset`] instead if these constraints are
414    /// difficult to satisfy. The only advantage of this method is that it
415    /// enables more aggressive compiler optimizations.
416    ///
417    /// [`wrapping_offset`]: #method.wrapping_offset
418    /// [allocated object]: crate::ptr#allocated-object
419    ///
420    /// # Examples
421    ///
422    /// ```
423    /// let s: &str = "123";
424    /// let ptr: *const u8 = s.as_ptr();
425    ///
426    /// unsafe {
427    ///     assert_eq!(*ptr.offset(1) as char, '2');
428    ///     assert_eq!(*ptr.offset(2) as char, '3');
429    /// }
430    /// ```
431    #[stable(feature = "rust1", since = "1.0.0")]
432    #[must_use = "returns a new pointer rather than modifying its argument"]
433    #[rustc_const_stable(feature = "const_ptr_offset", since = "1.61.0")]
434    #[inline(always)]
435    #[cfg_attr(miri, track_caller)] // even without panics, this helps for Miri backtraces
436    #[cfg(not(feature = "ferrocene_certified"))]
437    pub const unsafe fn offset(self, count: isize) -> *const T
438    where
439        T: Sized,
440    {
441        #[inline]
442        #[rustc_allow_const_fn_unstable(const_eval_select)]
443        const fn runtime_offset_nowrap(this: *const (), count: isize, size: usize) -> bool {
444            // We can use const_eval_select here because this is only for UB checks.
445            const_eval_select!(
446                @capture { this: *const (), count: isize, size: usize } -> bool:
447                if const {
448                    true
449                } else {
450                    // `size` is the size of a Rust type, so we know that
451                    // `size <= isize::MAX` and thus `as` cast here is not lossy.
452                    let Some(byte_offset) = count.checked_mul(size as isize) else {
453                        return false;
454                    };
455                    let (_, overflow) = this.addr().overflowing_add_signed(byte_offset);
456                    !overflow
457                }
458            )
459        }
460
461        ub_checks::assert_unsafe_precondition!(
462            check_language_ub,
463            "ptr::offset requires the address calculation to not overflow",
464            (
465                this: *const () = self as *const (),
466                count: isize = count,
467                size: usize = size_of::<T>(),
468            ) => runtime_offset_nowrap(this, count, size)
469        );
470
471        // SAFETY: the caller must uphold the safety contract for `offset`.
472        unsafe { intrinsics::offset(self, count) }
473    }

496    /// Adds a signed offset to a pointer using wrapping arithmetic.
497    ///
498    /// `count` is in units of T; e.g., a `count` of 3 represents a pointer
499    /// offset of `3 * size_of::<T>()` bytes.
500    ///
501    /// # Safety
502    ///
503    /// This operation itself is always safe, but using the resulting pointer is not.
504    ///
505    /// The resulting pointer "remembers" the [allocated object] that `self` points to
506    /// (this is called "[Provenance](ptr/index.html#provenance)").
507    /// The pointer must not be used to read or write other allocated objects.
508    ///
509    /// In other words, `let z = x.wrapping_offset((y as isize) - (x as isize))` does *not* make `z`
510    /// the same as `y` even if we assume `T` has size `1` and there is no overflow: `z` is still
511    /// attached to the object `x` is attached to, and dereferencing it is Undefined Behavior unless
512    /// `x` and `y` point into the same allocated object.
513    ///
514    /// Compared to [`offset`], this method basically delays the requirement of staying within the
515    /// same allocated object: [`offset`] is immediate Undefined Behavior when crossing object
516    /// boundaries; `wrapping_offset` produces a pointer but still leads to Undefined Behavior if a
517    /// pointer is dereferenced when it is out-of-bounds of the object it is attached to. [`offset`]
518    /// can be optimized better and is thus preferable in performance-sensitive code.
519    ///
520    /// The delayed check only considers the value of the pointer that was dereferenced, not the
521    /// intermediate values used during the computation of the final result. For example,
522    /// `x.wrapping_offset(o).wrapping_offset(o.wrapping_neg())` is always the same as `x`. In other
523    /// words, leaving the allocated object and then re-entering it later is permitted.
524    ///
525    /// [`offset`]: #method.offset
526    /// [allocated object]: crate::ptr#allocated-object
527    ///
528    /// # Examples
529    ///
530    /// ```
531    /// # use std::fmt::Write;
532    /// // Iterate using a raw pointer in increments of two elements
533    /// let data = [1u8, 2, 3, 4, 5];
534    /// let mut ptr: *const u8 = data.as_ptr();
535    /// let step = 2;
536    /// let end_rounded_up = ptr.wrapping_offset(6);
537    ///
538    /// let mut out = String::new();
539    /// while ptr != end_rounded_up {
540    ///     unsafe {

619    /// Calculates the distance between two pointers within the same allocation. The returned value is in
620    /// units of T: the distance in bytes divided by `size_of::<T>()`.
621    ///
622    /// This is equivalent to `(self as isize - origin as isize) / (size_of::<T>() as isize)`,
623    /// except that it has a lot more opportunities for UB, in exchange for the compiler
624    /// better understanding what you are doing.
625    ///
626    /// The primary motivation of this method is for computing the `len` of an array/slice
627    /// of `T` that you are currently representing as a "start" and "end" pointer
628    /// (and "end" is "one past the end" of the array).
629    /// In that case, `end.offset_from(start)` gets you the length of the array.
630    ///
631    /// All of the following safety requirements are trivially satisfied for this usecase.
632    ///
633    /// [`offset`]: #method.offset
634    ///
635    /// # Safety
636    ///
637    /// If any of the following conditions are violated, the result is Undefined Behavior:
638    ///
639    /// * `self` and `origin` must either
640    ///
641    ///   * point to the same address, or
642    ///   * both be [derived from][crate::ptr#provenance] a pointer to the same [allocated object], and the memory range between
643    ///     the two pointers must be in bounds of that object. (See below for an example.)
644    ///
645    /// * The distance between the pointers, in bytes, must be an exact multiple
646    ///   of the size of `T`.
647    ///
648    /// As a consequence, the absolute distance between the pointers, in bytes, computed on
649    /// mathematical integers (without "wrapping around"), cannot overflow an `isize`. This is
650    /// implied by the in-bounds requirement, and the fact that no allocated object can be larger
651    /// than `isize::MAX` bytes.
652    ///
653    /// The requirement for pointers to be derived from the same allocated object is primarily
654    /// needed for `const`-compatibility: the distance between pointers into *different* allocated
655    /// objects is not known at compile-time. However, the requirement also exists at
656    /// runtime and may be exploited by optimizations. If you wish to compute the difference between
657    /// pointers that are not guaranteed to be from the same allocation, use `(self as isize -
658    /// origin as isize) / size_of::<T>()`.
659    // FIXME: recommend `addr()` instead of `as usize` once that is stable.
660    ///
661    /// [`add`]: #method.add
662    /// [allocated object]: crate::ptr#allocated-object
663    ///
664    /// # Panics
665    ///
666    /// This function panics if `T` is a Zero-Sized Type ("ZST").
667    ///
668    /// # Examples
669    ///
670    /// Basic usage:
671    ///
672    /// ```
673    /// let a = [0; 5];
674    /// let ptr1: *const i32 = &a[1];
675    /// let ptr2: *const i32 = &a[3];
676    /// unsafe {
677    ///     assert_eq!(ptr2.offset_from(ptr1), 2);
678    ///     assert_eq!(ptr1.offset_from(ptr2), -2);
679    ///     assert_eq!(ptr1.offset(2), ptr2);
680    ///     assert_eq!(ptr2.offset(-2), ptr1);
681    /// }
682    /// ```
683    ///
684    /// *Incorrect* usage:
685    ///
686    /// ```rust,no_run
687    /// let ptr1 = Box::into_raw(Box::new(0u8)) as *const u8;
688    /// let ptr2 = Box::into_raw(Box::new(1u8)) as *const u8;
689    /// let diff = (ptr2 as isize).wrapping_sub(ptr1 as isize);
690    /// // Make ptr2_other an "alias" of ptr2.add(1), but derived from ptr1.
691    /// let ptr2_other = (ptr1 as *const u8).wrapping_offset(diff).wrapping_offset(1);
692    /// assert_eq!(ptr2 as usize, ptr2_other as usize);
693    /// // Since ptr2_other and ptr2 are derived from pointers to different objects,
694    /// // computing their offset is undefined behavior, even though
695    /// // they point to addresses that are in-bounds of the same object!
696    /// unsafe {
697    ///     let one = ptr2_other.offset_from(ptr2); // Undefined Behavior! 
698    /// }
699    /// ```

701    #[rustc_const_stable(feature = "const_ptr_offset_from", since = "1.65.0")]

767    /// a larger offset, it's still *not permitted* to pass pointers which differ

789    /// // ptr1.offset_from_unsigned(ptr2)
790    /// ```

1130    /// Adds an unsigned offset to a pointer using wrapping arithmetic.
1131    ///
1132    /// `count` is in units of T; e.g., a `count` of 3 represents a pointer
1133    /// offset of `3 * size_of::<T>()` bytes.
1134    ///
1135    /// # Safety
1136    ///
1137    /// This operation itself is always safe, but using the resulting pointer is not.
1138    ///
1139    /// The resulting pointer "remembers" the [allocated object] that `self` points to; it must not
1140    /// be used to read or write other allocated objects.
1141    ///
1142    /// In other words, `let z = x.wrapping_add((y as usize) - (x as usize))` does *not* make `z`
1143    /// the same as `y` even if we assume `T` has size `1` and there is no overflow: `z` is still
1144    /// attached to the object `x` is attached to, and dereferencing it is Undefined Behavior unless
1145    /// `x` and `y` point into the same allocated object.
1146    ///
1147    /// Compared to [`add`], this method basically delays the requirement of staying within the
1148    /// same allocated object: [`add`] is immediate Undefined Behavior when crossing object
1149    /// boundaries; `wrapping_add` produces a pointer but still leads to Undefined Behavior if a
1150    /// pointer is dereferenced when it is out-of-bounds of the object it is attached to. [`add`]
1151    /// can be optimized better and is thus preferable in performance-sensitive code.
1152    ///
1153    /// The delayed check only considers the value of the pointer that was dereferenced, not the
1154    /// intermediate values used during the computation of the final result. For example,
1155    /// `x.wrapping_add(o).wrapping_sub(o)` is always the same as `x`. In other words, leaving the
1156    /// allocated object and then re-entering it later is permitted.
1157    ///
1158    /// [`add`]: #method.add
1159    /// [allocated object]: crate::ptr#allocated-object
1160    ///
1161    /// # Examples

1168    /// let step = 2;
1169    /// let end_rounded_up = ptr.wrapping_add(6);
1170    ///

1603    /// is *[undefined behavior]* even if the resulting pointer is not used.
1604    ///
1605    /// [undefined behavior]: https://doc.rust-lang.org/reference/behavior-considered-undefined.html
1606    ///
1607    /// # Examples
1608    ///
1609    /// ```
1610    /// #![feature(slice_ptr_get)]
1611    ///
1612    /// let x = &[1, 2, 4] as *const [i32];
1613    ///
1614    /// unsafe {
1615    ///     assert_eq!(x.get_unchecked(1), x.as_ptr().add(1));
1616    /// }
1617    /// ```
1618    #[unstable(feature = "slice_ptr_get", issue = "74265")]
1619    #[inline]
1620    pub unsafe fn get_unchecked<I>(self, index: I) -> *const I::Output
1621    where
1622        I: SliceIndex<[T]>,
1623    {
1624        // SAFETY: the caller ensures that `self` is dereferenceable and `index` in-bounds.
1625        unsafe { index.get_unchecked(self) }
1626    }

1628    /// Returns `None` if the pointer is null, or else returns a shared slice to
1629    /// the value wrapped in `Some`. In contrast to [`as_ref`], this does not require
1630    /// that the value has to be initialized.
1631    ///
1632    /// [`as_ref`]: #method.as_ref
1633    ///
1634    /// # Safety
1635    ///
1636    /// When calling this method, you have to ensure that *either* the pointer is null *or*
1637    /// all of the following is true:
1638    ///
1639    /// * The pointer must be [valid] for reads for `ptr.len() * size_of::<T>()` many bytes,
1640    ///   and it must be properly aligned. This means in particular:
1641    ///
1642    ///     * The entire memory range of this slice must be contained within a single [allocated object]!
1643    ///       Slices can never span across multiple allocated objects.
1644    ///
1645    ///     * The pointer must be aligned even for zero-length slices. One
1646    ///       reason for this is that enum layout optimizations may rely on references
1647    ///       (including slices of any length) being aligned and non-null to distinguish
1648    ///       them from other data. You can obtain a pointer that is usable as `data`
1649    ///       for zero-length slices using [`NonNull::dangling()`].
1650    ///
1651    /// * The total size `ptr.len() * size_of::<T>()` of the slice must be no larger than `isize::MAX`.
1652    ///   See the safety documentation of [`pointer::offset`].
1653    ///
1654    /// * You must enforce Rust's aliasing rules, since the returned lifetime `'a` is
1655    ///   arbitrarily chosen and does not necessarily reflect the actual lifetime of the data.
1656    ///   In particular, while this reference exists, the memory the pointer points to must
1657    ///   not get mutated (except inside `UnsafeCell`).
1658    ///
1659    /// This applies even if the result of this method is unused!
1660    ///
1661    /// See also [`slice::from_raw_parts`][].
1662    ///
1663    /// [valid]: crate::ptr#safety
1664    /// [allocated object]: crate::ptr#allocated-object
1665    ///
1666    /// # Panics during const evaluation
1667    ///
1668    /// This method will panic during const evaluation if the pointer cannot be
1669    /// determined to be null or not. See [`is_null`] for more information.
1670    ///
1671    /// [`is_null`]: #method.is_null
1672    #[inline]
1673    #[unstable(feature = "ptr_as_uninit", issue = "75402")]
1674    pub const unsafe fn as_uninit_slice<'a>(self) -> Option<&'a [MaybeUninit<T>]> {
1675        if self.is_null() {
1676            None
1677        } else {
1678            // SAFETY: the caller must uphold the safety contract for `as_uninit_slice`.
1679            Some(unsafe { slice::from_raw_parts(self as *const MaybeUninit<T>, self.len()) })
1680        }
1681    }
1682}
