47            (left_val, right_val) => {

130/// Asserts that an expression matches the provided pattern.
131///
132/// This macro is generally preferable to `assert!(matches!(value, pattern))`, because it can print
133/// the debug representation of the actual value shape that did not meet expectations. In contrast,
134/// using [`assert!`] will only print that expectations were not met, but not why.
135///
136/// The pattern syntax is exactly the same as found in a match arm and the `matches!` macro. The
137/// optional if guard can be used to add additional checks that must be true for the matched value,
138/// otherwise this macro will panic.
139///
140/// Assertions are always checked in both debug and release builds, and cannot
141/// be disabled. See [`debug_assert_matches!`] for assertions that are disabled in
142/// release builds by default.
143///
144/// [`debug_assert_matches!`]: crate::assert_matches::debug_assert_matches
145///
146/// On panic, this macro will print the value of the expression with its debug representation.
147///
148/// Like [`assert!`], this macro has a second form, where a custom panic message can be provided.
149///
150/// # Examples
151///
152/// ```
153/// #![feature(assert_matches)]
154///
155/// use std::assert_matches::assert_matches;
156///
157/// let a = Some(345);
158/// let b = Some(56);
159/// assert_matches!(a, Some(_));
160/// assert_matches!(b, Some(_));
161///
162/// assert_matches!(a, Some(345));
163/// assert_matches!(a, Some(345) | None);
164///
165/// // assert_matches!(a, None); // panics
166/// // assert_matches!(b, Some(345)); // panics
167/// // assert_matches!(b, Some(345) | None); // panics
168///
169/// assert_matches!(a, Some(x) if x > 100);
170/// // assert_matches!(a, Some(x) if x < 100); // panics
171/// ```
172#[unstable(feature = "assert_matches", issue = "82775")]
173#[allow_internal_unstable(panic_internals)]
174#[rustc_macro_transparency = "semitransparent"]
175#[cfg(not(feature = "ferrocene_certified"))]
176pub macro assert_matches {
177    ($left:expr, $(|)? $( $pattern:pat_param )|+ $( if $guard: expr )? $(,)?) => {
178        match $left {
179            $( $pattern )|+ $( if $guard )? => {}
180            ref left_val => {
181                $crate::panicking::assert_matches_failed(
182                    left_val,
183                    $crate::stringify!($($pattern)|+ $(if $guard)?),
184                    $crate::option::Option::None
185                );
186            }
187        }
188    },
189    ($left:expr, $(|)? $( $pattern:pat_param )|+ $( if $guard: expr )?, $($arg:tt)+) => {
190        match $left {
191            $( $pattern )|+ $( if $guard )? => {}
192            ref left_val => {
193                $crate::panicking::assert_matches_failed(
194                    left_val,
195                    $crate::stringify!($($pattern)|+ $(if $guard)?),
196                    $crate::option::Option::Some($crate::format_args!($($arg)+))
197                );
198            }
199        }
200    },

203/// A macro for defining `#[cfg]` match-like statements.
204///
205/// It is similar to the `if/elif` C preprocessor macro by allowing definition of a cascade of
206/// `#[cfg]` cases, emitting the implementation which matches first.
207///
208/// This allows you to conveniently provide a long list `#[cfg]`'d blocks of code
209/// without having to rewrite each clause multiple times.
210///
211/// Trailing `_` wildcard match arms are **optional** and they indicate a fallback branch when
212/// all previous declarations do not evaluate to true.
213///
214/// # Example
215///
216/// ```
217/// #![feature(cfg_select)]
218///
219/// cfg_select! {
220///     unix => {
221///         fn foo() { /* unix specific functionality */ }
222///     }
223///     target_pointer_width = "32" => {
224///         fn foo() { /* non-unix, 32-bit functionality */ }
225///     }
226///     _ => {
227///         fn foo() { /* fallback implementation */ }
228///     }
229/// }
230/// ```
231///
232/// If desired, it is possible to return expressions through the use of surrounding braces:
233///
234/// ```
235/// #![feature(cfg_select)]
236///
237/// let _some_string = cfg_select! {{
238///     unix => { "With great power comes great electricity bills" }
239///     _ => { "Behind every successful diet is an unwatched pizza" }
240/// }};
241/// ```
242#[unstable(feature = "cfg_select", issue = "115585")]
243#[rustc_diagnostic_item = "cfg_select"]
244#[rustc_macro_transparency = "semitransparent"]
245#[cfg(not(feature = "ferrocene_certified"))]
246pub macro cfg_select {
247    ({ $($tt:tt)* }) => {{
248        $crate::cfg_select! { $($tt)* }
249    }},
250    (_ => { $($output:tt)* }) => {
251        $($output)*
252    },
253    (
254        $cfg:meta => $output:tt
255        $($( $rest:tt )+)?
256    ) => {
257        #[cfg($cfg)]
258        $crate::cfg_select! { _ => $output }
259        $(
260            #[cfg(not($cfg))]
261            $crate::cfg_select! { $($rest)+ }
262        )?
263    },
264}

266/// Asserts that a boolean expression is `true` at runtime.
267///
268/// This will invoke the [`panic!`] macro if the provided expression cannot be
269/// evaluated to `true` at runtime.
270///
271/// Like [`assert!`], this macro also has a second version, where a custom panic
272/// message can be provided.
273///
274/// # Uses
275///
276/// Unlike [`assert!`], `debug_assert!` statements are only enabled in non
277/// optimized builds by default. An optimized build will not execute
278/// `debug_assert!` statements unless `-C debug-assertions` is passed to the
279/// compiler. This makes `debug_assert!` useful for checks that are too
280/// expensive to be present in a release build but may be helpful during
281/// development. The result of expanding `debug_assert!` is always type checked.
282///
283/// An unchecked assertion allows a program in an inconsistent state to keep
284/// running, which might have unexpected consequences but does not introduce
285/// unsafety as long as this only happens in safe code. The performance cost
286/// of assertions, however, is not measurable in general. Replacing [`assert!`]
287/// with `debug_assert!` is thus only encouraged after thorough profiling, and
288/// more importantly, only in safe code!
289///
290/// # Examples
291///
292/// ```
293/// // the panic message for these assertions is the stringified value of the
294/// // expression given.
295/// debug_assert!(true);
296///
297/// fn some_expensive_computation() -> bool { true } // a very simple function
298/// debug_assert!(some_expensive_computation());
299///
300/// // assert with a custom message
301/// let x = true;
302/// debug_assert!(x, "x wasn't true!");
303///
304/// let a = 3; let b = 27;
305/// debug_assert!(a + b == 30, "a = {}, b = {}", a, b);
306/// ```
307#[macro_export]
308#[stable(feature = "rust1", since = "1.0.0")]
309#[rustc_diagnostic_item = "debug_assert_macro"]
310#[allow_internal_unstable(edition_panic)]
311#[cfg(not(feature = "ferrocene_certified"))]
312macro_rules! debug_assert {
313    ($($arg:tt)*) => {
314        if $crate::cfg!(debug_assertions) {
315            $crate::assert!($($arg)*);
316        }
317    };
318}

465/// Unwraps a result or propagates its error.
466///
467/// The [`?` operator][propagating-errors] was added to replace `try!`
468/// and should be used instead. Furthermore, `try` is a reserved word
469/// in Rust 2018, so if you must use it, you will need to use the
470/// [raw-identifier syntax][ris]: `r#try`.
471///
472/// [propagating-errors]: https://doc.rust-lang.org/book/ch09-02-recoverable-errors-with-result.html#a-shortcut-for-propagating-errors-the--operator
473/// [ris]: https://doc.rust-lang.org/nightly/rust-by-example/compatibility/raw_identifiers.html
474///
475/// `try!` matches the given [`Result`]. In case of the `Ok` variant, the
476/// expression has the value of the wrapped value.
477///
478/// In case of the `Err` variant, it retrieves the inner error. `try!` then
479/// performs conversion using `From`. This provides automatic conversion
480/// between specialized errors and more general ones. The resulting
481/// error is then immediately returned.
482///
483/// Because of the early return, `try!` can only be used in functions that
484/// return [`Result`].
485///
486/// # Examples
487///
488/// ```
489/// use std::io;
490/// use std::fs::File;
491/// use std::io::prelude::*;
492///
493/// enum MyError {
494///     FileWriteError
495/// }
496///
497/// impl From<io::Error> for MyError {
498///     fn from(e: io::Error) -> MyError {
499///         MyError::FileWriteError
500///     }
501/// }
502///
503/// // The preferred method of quick returning Errors
504/// fn write_to_file_question() -> Result<(), MyError> {
505///     let mut file = File::create("my_best_friends.txt")?;
506///     file.write_all(b"This is a list of my best friends.")?;
507///     Ok(())
508/// }
509///
510/// // The previous method of quick returning Errors
511/// fn write_to_file_using_try() -> Result<(), MyError> {
512///     let mut file = r#try!(File::create("my_best_friends.txt"));
513///     r#try!(file.write_all(b"This is a list of my best friends."));
514///     Ok(())
515/// }
516///
517/// // This is equivalent to:
518/// fn write_to_file_using_match() -> Result<(), MyError> {
519///     let mut file = r#try!(File::create("my_best_friends.txt"));
520///     match file.write_all(b"This is a list of my best friends.") {
521///         Ok(v) => v,
522///         Err(e) => return Err(From::from(e)),
523///     }
524///     Ok(())
525/// }
526/// ```
527#[macro_export]
528#[stable(feature = "rust1", since = "1.0.0")]
529#[deprecated(since = "1.39.0", note = "use the `?` operator instead")]
530#[doc(alias = "?")]
531#[cfg(not(feature = "ferrocene_certified"))]
532macro_rules! r#try {
533    ($expr:expr $(,)?) => {
534        match $expr {
535            $crate::result::Result::Ok(val) => val,
536            $crate::result::Result::Err(err) => {
537                return $crate::result::Result::Err($crate::convert::From::from(err));
538            }
539        }
540    };
541}

543/// Writes formatted data into a buffer.
544///
545/// This macro accepts a 'writer', a format string, and a list of arguments. Arguments will be
546/// formatted according to the specified format string and the result will be passed to the writer.
547/// The writer may be any value with a `write_fmt` method; generally this comes from an
548/// implementation of either the [`fmt::Write`] or the [`io::Write`] trait. The macro
549/// returns whatever the `write_fmt` method returns; commonly a [`fmt::Result`], or an
550/// [`io::Result`].
551///
552/// See [`std::fmt`] for more information on the format string syntax.
553///
554/// [`std::fmt`]: ../std/fmt/index.html
555/// [`fmt::Write`]: crate::fmt::Write
556/// [`io::Write`]: ../std/io/trait.Write.html
557/// [`fmt::Result`]: crate::fmt::Result
558/// [`io::Result`]: ../std/io/type.Result.html
559///
560/// # Examples
561///
562/// ```
563/// use std::io::Write;
564///
565/// fn main() -> std::io::Result<()> {
566///     let mut w = Vec::new();
567///     write!(&mut w, "test")?;
568///     write!(&mut w, "formatted {}", "arguments")?;
569///
570///     assert_eq!(w, b"testformatted arguments");
571///     Ok(())
572/// }
573/// ```
574///
575/// A module can import both `std::fmt::Write` and `std::io::Write` and call `write!` on objects
576/// implementing either, as objects do not typically implement both. However, the module must
577/// avoid conflict between the trait names, such as by importing them as `_` or otherwise renaming
578/// them:
579///
580/// ```
581/// use std::fmt::Write as _;
582/// use std::io::Write as _;
583///
584/// fn main() -> Result<(), Box<dyn std::error::Error>> {
585///     let mut s = String::new();
586///     let mut v = Vec::new();
587///
588///     write!(&mut s, "{} {}", "abc", 123)?; // uses fmt::Write::write_fmt
589///     write!(&mut v, "s = {:?}", s)?; // uses io::Write::write_fmt
590///     assert_eq!(v, b"s = \"abc 123\"");
591///     Ok(())
592/// }
593/// ```
594///
595/// If you also need the trait names themselves, such as to implement one or both on your types,
596/// import the containing module and then name them with a prefix:
597///
598/// ```
599/// # #![allow(unused_imports)]
600/// use std::fmt::{self, Write as _};
601/// use std::io::{self, Write as _};
602///
603/// struct Example;
604///
605/// impl fmt::Write for Example {
606///     fn write_str(&mut self, _s: &str) -> core::fmt::Result {
607///          unimplemented!();
608///     }
609/// }
610/// ```
611///
612/// Note: This macro can be used in `no_std` setups as well.
613/// In a `no_std` setup you are responsible for the implementation details of the components.
614///
615/// ```no_run
616/// use core::fmt::Write;
617///
618/// struct Example;
619///
620/// impl Write for Example {
621///     fn write_str(&mut self, _s: &str) -> core::fmt::Result {
622///          unimplemented!();
623///     }
624/// }
625///
626/// let mut m = Example{};
627/// write!(&mut m, "Hello World").expect("Not written");
628/// ```
629#[macro_export]
630#[stable(feature = "rust1", since = "1.0.0")]
631#[rustc_diagnostic_item = "write_macro"]
632#[cfg(not(feature = "ferrocene_certified"))]
633macro_rules! write {
634    ($dst:expr, $($arg:tt)*) => {
635        $dst.write_fmt($crate::format_args!($($arg)*))
636    };
637}

650///
651/// ```

654/// fn main() -> Result<()> {
655///     let mut w = Vec::new();

657///     writeln!(&mut w, "test")?;
658///     writeln!(&mut w, "formatted {}", "arguments")?;

678/// Indicates unreachable code.
679///
680/// This is useful any time that the compiler can't determine that some code is unreachable. For
681/// example:
682///
683/// * Match arms with guard conditions.
684/// * Loops that dynamically terminate.
685/// * Iterators that dynamically terminate.
686///
687/// If the determination that the code is unreachable proves incorrect, the
688/// program immediately terminates with a [`panic!`].
689///
690/// The unsafe counterpart of this macro is the [`unreachable_unchecked`] function, which
691/// will cause undefined behavior if the code is reached.
692///
693/// [`unreachable_unchecked`]: crate::hint::unreachable_unchecked
694///
695/// # Panics
696///
697/// This will always [`panic!`] because `unreachable!` is just a shorthand for `panic!` with a
698/// fixed, specific message.
699///
700/// Like `panic!`, this macro has a second form for displaying custom values.
701///
702/// # Examples
703///
704/// Match arms:
705///
706/// ```
707/// # #[allow(dead_code)]
708/// fn foo(x: Option<i32>) {
709///     match x {
710///         Some(n) if n >= 0 => println!("Some(Non-negative)"),
711///         Some(n) if n <  0 => println!("Some(Negative)"),
712///         Some(_)           => unreachable!(), // compile error if commented out
713///         None              => println!("None")
714///     }
715/// }
716/// ```
717///
718/// Iterators:
719///
720/// ```
721/// # #[allow(dead_code)]
722/// fn divide_by_three(x: u32) -> u32 { // one of the poorest implementations of x/3
723///     for i in 0.. {
724///         if 3*i < i { panic!("u32 overflow"); }
725///         if x < 3*i { return i-1; }
726///     }
727///     unreachable!("The loop should always return");
728/// }
729/// ```

732#[allow_internal_unstable(edition_panic)]

735#[cfg(not(feature = "ferrocene_certified"))]
736macro_rules! unreachable {
737    // Expands to either `$crate::panic::unreachable_2015` or `$crate::panic::unreachable_2021`
738    // depending on the edition of the caller.
739    ($($arg:tt)*) => {
740        /* compiler built-in */
741    };
742}

744/// Indicates unimplemented code by panicking with a message of "not implemented".
745///
746/// This allows your code to type-check, which is useful if you are prototyping or
747/// implementing a trait that requires multiple methods which you don't plan to use all of.
748///
749/// The difference between `unimplemented!` and [`todo!`] is that while `todo!`
750/// conveys an intent of implementing the functionality later and the message is "not yet
751/// implemented", `unimplemented!` makes no such claims. Its message is "not implemented".
752///
753/// Also, some IDEs will mark `todo!`s.
754///
755/// # Panics
756///
757/// This will always [`panic!`] because `unimplemented!` is just a shorthand for `panic!` with a
758/// fixed, specific message.
759///
760/// Like `panic!`, this macro has a second form for displaying custom values.
761///
762/// [`todo!`]: crate::todo
763///
764/// # Examples
765///
766/// Say we have a trait `Foo`:
767///
768/// ```
769/// trait Foo {
770///     fn bar(&self) -> u8;
771///     fn baz(&self);
772///     fn qux(&self) -> Result<u64, ()>;
773/// }
774/// ```
775///
776/// We want to implement `Foo` for 'MyStruct', but for some reason it only makes sense
777/// to implement the `bar()` function. `baz()` and `qux()` will still need to be defined
778/// in our implementation of `Foo`, but we can use `unimplemented!` in their definitions
779/// to allow our code to compile.
780///
781/// We still want to have our program stop running if the unimplemented methods are
782/// reached.
783///
784/// ```
785/// # trait Foo {
786/// #     fn bar(&self) -> u8;
787/// #     fn baz(&self);
788/// #     fn qux(&self) -> Result<u64, ()>;
789/// # }
790/// struct MyStruct;
791///
792/// impl Foo for MyStruct {
793///     fn bar(&self) -> u8 {
794///         1 + 1
795///     }
796///
797///     fn baz(&self) {
798///         // It makes no sense to `baz` a `MyStruct`, so we have no logic here
799///         // at all.
800///         // This will display "thread 'main' panicked at 'not implemented'".
801///         unimplemented!();
802///     }
803///
804///     fn qux(&self) -> Result<u64, ()> {
805///         // We have some logic here,
806///         // We can add a message to unimplemented! to display our omission.
807///         // This will display:
808///         // "thread 'main' panicked at 'not implemented: MyStruct isn't quxable'".
809///         unimplemented!("MyStruct isn't quxable");
810///     }
811/// }
812///
813/// fn main() {
814///     let s = MyStruct;
815///     s.bar();
816/// }
817/// ```
818#[macro_export]
819#[stable(feature = "rust1", since = "1.0.0")]
820#[rustc_diagnostic_item = "unimplemented_macro"]
821#[allow_internal_unstable(panic_internals)]
822#[cfg(not(feature = "ferrocene_certified"))]
823macro_rules! unimplemented {
824    () => {
825        $crate::panicking::panic("not implemented")
826    };
827    ($($arg:tt)+) => {
828        $crate::panic!("not implemented: {}", $crate::format_args!($($arg)+))
829    };
830}

835/// want a placeholder to let your code pass type analysis.

838/// an intent of implementing the functionality later and the message is "not yet

840///
841/// Also, some IDEs will mark `todo!`s.
842///

844///
845/// This will always [`panic!`] because `todo!` is just a shorthand for `panic!` with a
846/// fixed, specific message.
847///
848/// Like `panic!`, this macro has a second form for displaying custom values.
849///
850/// # Examples
851///
852/// Here's an example of some in-progress code. We have a trait `Foo`:
853///
854/// ```
855/// trait Foo {
856///     fn bar(&self) -> u8;
857///     fn baz(&self);
858///     fn qux(&self) -> Result<u64, ()>;
859/// }

864/// `baz()` and `qux()`, so we can use `todo!`:
865///
866/// ```
867/// # trait Foo {
868/// #     fn bar(&self) -> u8;
869/// #     fn baz(&self);
870/// #     fn qux(&self) -> Result<u64, ()>;

875///     fn bar(&self) -> u8 {
876///         1 + 1

880///         // Let's not worry about implementing baz() for now
881///         todo!();
882///     }
883///
884///     fn qux(&self) -> Result<u64, ()> {
885///         // We can add a message to todo! to display our omission.
886///         // This will display:
