9/// A type storing a `usize` which is a power of two, and thus
10/// represents a possible alignment in the Rust abstract machine.
11///
12/// Note that particularly large alignments, while representable in this type,
13/// are likely not to be supported by actual allocators and linkers.
14#[unstable(feature = "ptr_alignment_type", issue = "102070")]
15#[cfg_attr(not(feature = "ferrocene_certified"), derive(Copy, Clone, PartialEq, Eq))]
16#[cfg_attr(feature = "ferrocene_certified", derive(Copy, Clone))]
17#[repr(transparent)]
18pub struct Alignment(AlignmentEnum);
19
20// Alignment is `repr(usize)`, but via extra steps.
21#[cfg(not(feature = "ferrocene_certified"))]
22const _: () = assert!(size_of::<Alignment>() == size_of::<usize>());
23#[cfg(not(feature = "ferrocene_certified"))]
24const _: () = assert!(align_of::<Alignment>() == align_of::<usize>());
25
26#[cfg(not(feature = "ferrocene_certified"))]
27fn _alignment_can_be_structurally_matched(a: Alignment) -> bool {
28    matches!(a, Alignment::MIN)
29}

31impl Alignment {
32    /// The smallest possible alignment, 1.
33    ///
34    /// All addresses are always aligned at least this much.
35    ///
36    /// # Examples
37    ///
38    /// ```
39    /// #![feature(ptr_alignment_type)]
40    /// use std::ptr::Alignment;
41    ///
42    /// assert_eq!(Alignment::MIN.as_usize(), 1);
43    /// ```
44    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
45    #[cfg(not(feature = "ferrocene_certified"))]
46    pub const MIN: Self = Self(AlignmentEnum::_Align1Shl0);
47
48    /// Returns the alignment for a type.
49    ///
50    /// This provides the same numerical value as [`align_of`],
51    /// but in an `Alignment` instead of a `usize`.
52    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
53    #[inline]
54    #[must_use]
55    #[cfg(not(feature = "ferrocene_certified"))]
56    pub const fn of<T>() -> Self {
57        // This can't actually panic since type alignment is always a power of two.
58        const { Alignment::new(align_of::<T>()).unwrap() }
59    }
60
61    /// Creates an `Alignment` from a `usize`, or returns `None` if it's
62    /// not a power of two.
63    ///
64    /// Note that `0` is not a power of two, nor a valid alignment.
65    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
66    #[inline]
67    pub const fn new(align: usize) -> Option<Self> {
68        if align.is_power_of_two() {
69            // SAFETY: Just checked it only has one bit set
70            Some(unsafe { Self::new_unchecked(align) })
71        } else {
72            None
73        }
74    }
75
76    /// Creates an `Alignment` from a power-of-two `usize`.
77    ///
78    /// # Safety
79    ///
80    /// `align` must be a power of two.
81    ///
82    /// Equivalently, it must be `1 << exp` for some `exp` in `0..usize::BITS`.
83    /// It must *not* be zero.
84    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
85    #[inline]
86    #[track_caller]
87    pub const unsafe fn new_unchecked(align: usize) -> Self {
88        assert_unsafe_precondition!(
89            check_language_ub,
90            "Alignment::new_unchecked requires a power of two",
91            (align: usize = align) => align.is_power_of_two()
92        );
93
94        // SAFETY: By precondition, this must be a power of two, and
95        // our variants encompass all possible powers of two.
96        unsafe { mem::transmute::<usize, Alignment>(align) }
97    }
98
99    /// Returns the alignment as a [`usize`].
100    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
101    #[inline]
102    pub const fn as_usize(self) -> usize {
103        self.0 as usize
104    }
105
106    /// Returns the alignment as a <code>[NonZero]<[usize]></code>.
107    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
108    #[inline]
109    #[cfg(not(feature = "ferrocene_certified"))]
110    pub const fn as_nonzero(self) -> NonZero<usize> {
111        // This transmutes directly to avoid the UbCheck in `NonZero::new_unchecked`
112        // since there's no way for the user to trip that check anyway -- the
113        // validity invariant of the type would have to have been broken earlier --
114        // and emitting it in an otherwise simple method is bad for compile time.
115
116        // SAFETY: All the discriminants are non-zero.
117        unsafe { mem::transmute::<Alignment, NonZero<usize>>(self) }
118    }
119
120    /// Returns the base-2 logarithm of the alignment.
121    ///
122    /// This is always exact, as `self` represents a power of two.
123    ///
124    /// # Examples
125    ///
126    /// ```
127    /// #![feature(ptr_alignment_type)]
128    /// use std::ptr::Alignment;
129    ///
130    /// assert_eq!(Alignment::of::<u8>().log2(), 0);
131    /// assert_eq!(Alignment::new(1024).unwrap().log2(), 10);
132    /// ```
133    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
134    #[inline]
135    #[cfg(not(feature = "ferrocene_certified"))]
136    pub const fn log2(self) -> u32 {
137        self.as_nonzero().trailing_zeros()
138    }
139
140    /// Returns a bit mask that can be used to match this alignment.
141    ///
142    /// This is equivalent to `!(self.as_usize() - 1)`.
143    ///
144    /// # Examples
145    ///
146    /// ```
147    /// #![feature(ptr_alignment_type)]
148    /// #![feature(ptr_mask)]
149    /// use std::ptr::{Alignment, NonNull};
150    ///
151    /// #[repr(align(1))] struct Align1(u8);
152    /// #[repr(align(2))] struct Align2(u16);
153    /// #[repr(align(4))] struct Align4(u32);
154    /// let one = <NonNull<Align1>>::dangling().as_ptr();
155    /// let two = <NonNull<Align2>>::dangling().as_ptr();
156    /// let four = <NonNull<Align4>>::dangling().as_ptr();
157    ///
158    /// assert_eq!(four.mask(Alignment::of::<Align1>().mask()), four);
159    /// assert_eq!(four.mask(Alignment::of::<Align2>().mask()), four);
160    /// assert_eq!(four.mask(Alignment::of::<Align4>().mask()), four);
161    /// assert_ne!(one.mask(Alignment::of::<Align4>().mask()), one);
162    /// ```
163    #[unstable(feature = "ptr_alignment_type", issue = "102070")]
164    #[inline]
165    #[cfg(not(feature = "ferrocene_certified"))]
166    pub const fn mask(self) -> usize {
167        // SAFETY: The alignment is always nonzero, and therefore decrementing won't overflow.
168        !(unsafe { self.as_usize().unchecked_sub(1) })
169    }
170
171    // FIXME(const-hack) Remove me once `Ord::max` is usable in const
172    #[cfg(not(feature = "ferrocene_certified"))]
173    pub(crate) const fn max(a: Self, b: Self) -> Self {
174        if a.as_usize() > b.as_usize() { a } else { b }
175    }
176}
